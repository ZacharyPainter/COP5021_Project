801 /* Add up the differences from p of the `surround' points
802 before p. */
803 in.dx = in.dy = in.dz = 0.0;
804
805 in = Vadd(in, Psubtract(CURVE_POINT(curve, prev)((curve)->point_list[prev].coord), candidate));
806 if (prevprev >= 0)
807 in = Vadd(in, Psubtract(CURVE_POINT(curve, prevprev)((curve)->point_list[prevprev].coord), candidate));
808
809 /* And the points after p. Don't use more points after p than we
810 ended up with before it. */
811 out.dx = out.dy = out.dz = 0.0;
812
813 out = Vadd(out, Psubtract(CURVE_POINT(curve, next)((curve)->point_list[next].coord), candidate));
814 if (nextnext < CURVE_LENGTH(curve)((curve)->length))
815 out = Vadd(out, Psubtract(CURVE_POINT(curve, nextnext)((curve)->point_list[nextnext].coord), candidate));
816
817 /* Start with the old point. */
818 new_point = candidate;
819 sum = Vadd(in, out);
820 /* We added 2*n+2 points, so we have to divide the sum by 2*n+2 */
821 new_point.x += sum.dx / 6;
822 new_point.y += sum.dy / 6;
823 new_point.z += sum.dz / 6;
824 if (fabs(prev_new_point.x - new_point.x) < 0.3 && fabs(prev_new_point.y - new_point.y) < 0.3 && fabs(prev_new_point.z - new_point.z) < 0.3) {
825 collapsed = TRUE(!(0));
826 break;
827 }
828
829 /* Put the newly computed point into a separate curve, so it
830 doesn't affect future computation (on this iteration). */
831 append_point(newcurve, prev_new_point = new_point);
832 }
833
834 if (collapsed)
835 free_curve(newcurve);
836 else {
837 /* Just as with the first point, we have to keep the last point. */
838 if (offset)
839 append_point(newcurve, LAST_CURVE_POINT(curve)((curve)->point_list[(curve)->length-1].coord));
840
841 /* Set the original curve to the newly filtered one, and go again. */
842 free_curve(curve);
843 *curve = *newcurve;
844 }
845 free(newcurve);
846 }
847
848 if (logging)
849 log_curve(curve, FALSE(0));
850}
