82 float fdx;

105 *ky = fdx;

776/* Return a Path consisting of just the corner points of the single path in a PathVector. If the
777PathVector has more than one path, or that one path is open, or any of its segments are curved, then the
778returned PathVector is an empty path. If the input path is already just straight lines and vertices the output will be the
779same as the sole path in the input. */
780
781Geom::Path PrintEmf::pathv_to_simple_polygon(Geom::PathVector const &pathv, int *vertices)
782{
783 Geom::Point P1_trail;
784 Geom::Point P1;
785 Geom::Point P1_lead;
786 Geom::Point v1,v2;
787 Geom::Path output;
788 Geom::Path bad;
789 Geom::PathVector pv = pathv_to_linear_and_cubic_beziers(pathv);
790 Geom::PathVector::const_iterator pit = pv.begin();
791 Geom::PathVector::const_iterator pit2 = pv.begin();
792 int first_seg=1;
793 ++pit2;
794 *vertices = 0;
795 if(pit->end_closed() != pit->end_default())return(bad); // path must be closed
796 if(pit2 != pv.end())return(bad); // there may only be one path
797 P1_trail = pit->finalPoint();
798 Geom::Path::const_iterator cit = pit->begin();
799 P1 = cit->initialPoint();
800 for(;cit != pit->end_closed();++cit) {
801 if (!is_straight_curve(*cit)) {
802 *vertices = 0;
803 return(bad);
804 }
805 P1_lead = cit->finalPoint();
806 if(Geom::are_near(P1_lead, P1, 1e-5))continue; // duplicate points at the same coordinate
807 v1 = unit_vector(P1 - P1_trail);
808 v2 = unit_vector(P1_lead - P1 );
809 if(Geom::are_near(dot(v1,v2), 1.0, 1e-5)){ // P1 is within a straight line
810 P1 = P1_lead;
811 continue;
812 }
813 // P1 is the center point of a turn of some angle
814 if(!*vertices){
815 output.start( P1 );
816 output.close( pit->closed() );
817 }
818 if(!Geom::are_near(P1, P1_trail, 1e-5)){ // possible for P1 to start on the end point
819 Geom::LineSegment ls(P1_trail, P1);
820 output.append(ls);
821 if(first_seg){
822 *vertices += 2;
823 first_seg=0;
824 }
825 else {
826 *vertices += 1;
827 }
828 }
829 P1_trail = P1;
830 P1 = P1_lead;
831 }
832 return(output);
833}

835/* Returns the simplified PathVector (no matter what).
836 Sets is_rect if it is a rectangle.
837 Sets angle that will rotate side closest to horizontal onto horizontal.
838*/
839Geom::Path PrintEmf::pathv_to_rect(Geom::PathVector const &pathv, bool *is_rect, double *angle)
840{
841 Geom::Point P1_trail;
842 Geom::Point P1;
843 Geom::Point v1,v2;
844 int vertices;
845 Geom::Path pR = pathv_to_simple_polygon(pathv, &vertices);
846 *is_rect = false;
847 if(vertices==4){ // or else it cannot be a rectangle
848 int vertex_count=0;
849 /* Get the ends of the LAST line segment.
850 Find minimum rotation to align rectangle with X,Y axes. (Very degenerate if it is rotated 45 degrees.) */
851 *angle = 10.0; /* must be > than the actual angle in radians. */
852 for(Geom::Path::iterator cit = pR.begin();; ++cit){
853 P1_trail = cit->initialPoint();
854 P1 = cit->finalPoint();
855 v1 = unit_vector(P1 - P1_trail);
856 if(v1[Geom::X] > 0){ // only check the 1 or 2 points on vectors aimed the same direction as unit X
857 double ang = asin(v1[Geom::Y]); // because component is rotation by ang of {1,0| vector
858 if(fabs(ang) < fabs(*angle))*angle = -ang; // y increases down, flips sign on angle
859 }
860 if(cit == pR.end_open())break;
861 }
862
863 /* For increased numerical stability, snap the angle to the nearest 1/100th of a degree. */
864 double convert = 36000.0/ (2.0 * M_PI3.14159265358979323846);
865 *angle = round(*angle * convert)/convert;
866
867 // at this stage v1 holds the last vector in the path, whichever direction it points.
868 for(Geom::Path::iterator cit = pR.begin(); ;++cit) {
869 v2 = v1;
870 P1_trail = cit->initialPoint();
871 P1 = cit->finalPoint();
872 v1 = unit_vector(P1 - P1_trail);
873 // P1 is center of a turn that is not 90 degrees. Limit comes from cos(89.9) = .001745
874 if(!Geom::are_near(dot(v1,v2), 0.0, 2e-3))break;
875 vertex_count++;
876 if(cit == pR.end_open())break;
877 }
878 if(vertex_count == 4){
879 *is_rect=true;
880 }
881 }
882 return(pR);
883}

885/* Compare a vector with a rectangle's orientation (angle needed to rotate side(s)
886 closest to horizontal to exactly horizontal) and return:
887 0 none of the following
888 1 parallel to horizontal
889 2 parallel to vertical
890 3 antiparallel to horizontal
891 4 antiparallel to vertical
892*/
893int PrintEmf::vector_rect_alignment(double angle, Geom::Point vtest){
894 int stat = 0;
895 Geom::Point v1 = Geom::unit_vector(vtest); // unit vector to test alignment
896 Geom::Point v2 = Geom::Point(1,0) * Geom::Rotate(-angle); // unit horizontal side (sign change because Y increases DOWN)
897 Geom::Point v3 = Geom::Point(0,1) * Geom::Rotate(-angle); // unit horizontal side (sign change because Y increases DOWN)
898 if( Geom::are_near(dot(v1,v2), 1.0, 1e-5)){ stat = 1; }
899 else if(Geom::are_near(dot(v1,v2),-1.0, 1e-5)){ stat = 2; }
900 else if(Geom::are_near(dot(v1,v3), 1.0, 1e-5)){ stat = 3; }
901 else if(Geom::are_near(dot(v1,v3),-1.0, 1e-5)){ stat = 4; }
902 return(stat);
903}

906 0 UL (and default)
907 1 UR
908 2 LR
909 3 LL
910 Needed because the start can be any point, and the direction could run either
911 clockwise or counterclockwise. This should work even if the corners of the rectangle
912 are slightly displaced.
913*/
914Geom::Point PrintEmf::get_pathrect_corner(Geom::Path pathRect, double angle, int corner){
915 Geom::Point center(0,0);

917 center += cit->initialPoint()/4.0;

922 int UL; // 1 if Lower, 0 if Upper (as viewed on screen, y coordinates increase downwards)
923 switch(corner){
924 case 1: //UR
925 LR = 0;
926 UL = 0;
927 break;
928 case 2: //LR
929 LR = 0;
930 UL = 1;
931 break;
932 case 3: //LL
933 LR = 1;
934 UL = 1;
935 break;
936 default: //UL
937 LR = 1;
938 UL = 0;
939 break;
940 }

942 Geom::Point v1 = Geom::Point(1,0) * Geom::Rotate(-angle); // unit horizontal side (sign change because Y increases DOWN)
943 Geom::Point v2 = Geom::Point(0,1) * Geom::Rotate(-angle); // unit vertical side (sign change because Y increases DOWN)
944 Geom::Point P1;
945 for(Geom::Path::iterator cit = pathRect.begin(); ; ++cit) {
946 P1 = cit->initialPoint();
947

949 && ( UL == (dot(P1 - center,v2) > 0 ? 1 : 0) ) ) break;
950 if(cit == pathRect.end_open())break;
951 }

969 same as the preceding one, leave the preceding one active. If style is NULL
970 terminate the current clip, if any, and return.
971*/
972void PrintEmf::do_clip_if_present(SPStyle const *style){
973 char *rec;
974 static SPClipPath *scpActive = nullptr;
975 if(!style){
976 if(scpActive){ // clear the existing clip
977 rec = U_EMRRESTOREDC_set(-1);
978 if (!rec || emf_append((PU_ENHMETARECORD)rec, et, U_REC_FREE1)) {
979 g_error("Fatal programming error in PrintEmf::fill at U_EMRRESTOREDC_set");
980 }
981 scpActive=nullptr;
982 }
983 } else {
984 /* The current implementation converts only one level of clipping. If there were more
985 clips further up the stack they should be combined with the pathvector using "and". Since this
986 comes up rarely, and would involve a lot of searching (all the way up the stack for every
987 draw operation), it has not yet been implemented.
988
989 Note, to debug this section of code use print statements on sp_svg_write_path(combined_pathvector).
990 */
991 /* find the first clip_ref at object or up the stack. There may not be one. */
992 SPClipPath *scp = nullptr;
993 SPItem *item = SP_ITEM(style->object);
994 while(true) {
995 scp = item->getClipObject();
996 if(scp)break;
997 item = SP_ITEM(item->parent);
998 if(!item || SP_IS_ROOT(item))break; // this will never be a clipping path
999 }
1000
1001 if(scp != scpActive){ // change or remove the clipping
1002 if(scpActive){ // clear the existing clip
1003 rec = U_EMRRESTOREDC_set(-1);
1004 if (!rec || emf_append((PU_ENHMETARECORD)rec, et, U_REC_FREE1)) {
1005 g_error("Fatal programming error in PrintEmf::fill at U_EMRRESTOREDC_set");
1006 }
1007 scpActive = nullptr;
1008 }
1009
1010 if (scp) { // set the new clip
1011 /* because of units and who knows what other transforms that might be applied above we
1012 need the full transform all the way to the root.
1013 */
1014 Geom::Affine tf = item->transform;
1015 SPItem *scan_item = item;
1016 while(true) {
1017 scan_item = SP_ITEM(scan_item->parent);
1018 if(!scan_item)break;
1019 tf *= scan_item->transform;
1020 }
1021 tf *= Geom::Scale(_doc_unit_scale);; // Transform must be in PIXELS, no matter what the document unit is.
1022
1023 /* find the clipping path */
1024 Geom::PathVector combined_pathvector;
1025 Geom::Affine tfc; // clipping transform, generally not the same as item transform
1026 for (auto& child: scp->children) {
1027 item = SP_ITEM(&child);
1028 if (!item) {
1029 break;
1030 }
1031 if (SP_IS_GROUP(item)) { // not implemented
1032 // return sp_group_render(item);
1033 combined_pathvector = merge_PathVector_with_group(combined_pathvector, item, tfc);
1034 } else if (SP_IS_SHAPE(item)) {
1035 combined_pathvector = merge_PathVector_with_shape(combined_pathvector, item, tfc);
1036 } else { // not implemented
1037 }
1038 }
1039
1040 if (!combined_pathvector.empty()) { // if clipping path isn't empty, define EMF clipping record
1041 scpActive = scp; // remember for next time
1042 // the sole purpose of this SAVEDC is to let us clear the clipping region later.
1043 rec = U_EMRSAVEDC_set();
1044 if (!rec || emf_append((PU_ENHMETARECORD)rec, et, U_REC_FREE1)) {
1045 g_error("Fatal programming error in PrintEmf::image at U_EMRSAVEDC_set");
1046 }
1047 (void) draw_pathv_to_EMF(combined_pathvector, tf);
1048 rec = U_EMRSELECTCLIPPATH_set(U_RGN_COPY5);
1049 if (!rec || emf_append((PU_ENHMETARECORD)rec, et, U_REC_FREE1)) {
1050 g_error("Fatal programming error in PrintEmf::do_clip_if_present at U_EMRSELECTCLIPPATH_set");
1051 }
1052 }
1053 else {
1054 scpActive = nullptr; // no valid path available to draw, so no DC was saved, so no signal to restore
1055 }
1056 } // change or remove clipping
1057 } // scp exists
1058 } // style exists
1059}

1061Geom::PathVector PrintEmf::merge_PathVector_with_group(Geom::PathVector const &combined_pathvector, SPItem const *item, const Geom::Affine &transform)
1062{
1063 // sanity test, only a group should be passed in, return empty if something else happens
1064 auto group = dynamic_cast<SPGroup const *>(item);
1065 if (!group)
1066 return {};
1067
1068 Geom::PathVector new_combined_pathvector = combined_pathvector;
1069 Geom::Affine tfc = item->transform * transform;
1070 for (auto& child: group->children) {
1071 item = SP_ITEM(&child);
1072 if (!item) {
1073 break;
1074 }
1075 if (SP_IS_GROUP(item)) {
1076 new_combined_pathvector = merge_PathVector_with_group(new_combined_pathvector, item, tfc); // could be endlessly recursive on a badly formed SVG
1077 } else if (SP_IS_SHAPE(item)) {
1078 new_combined_pathvector = merge_PathVector_with_shape(new_combined_pathvector, item, tfc);
1079 } else { // not implemented
1080 }
1081 }
1082 return new_combined_pathvector;
1083}

1085Geom::PathVector PrintEmf::merge_PathVector_with_shape(Geom::PathVector const &combined_pathvector, SPItem const *item, const Geom::Affine &transform)
1086{
1087 Geom::PathVector new_combined_pathvector;
1088 auto shape = dynamic_cast<SPShape const *>(item);
1089
1090 // sanity test, only a shape should be passed in, return empty if something else happens
1091 if (!shape)
1092 return new_combined_pathvector;
1093
1094 Geom::Affine tfc = item->transform * transform;
1095 if (shape->curve()) {
1096 Geom::PathVector const &new_vect = shape->curve()->get_pathvector();
1097 if(combined_pathvector.empty()){
1098 new_combined_pathvector = new_vect * tfc;
1099 }
1100 else {
1101 new_combined_pathvector = sp_pathvector_boolop(new_vect * tfc, combined_pathvector, bool_op_union , (FillRule) fill_oddEven, (FillRule) fill_oddEven);
1102 }
1103 }
1104 return new_combined_pathvector;
1105}

opa);

1376 } else {

1380 PPT, and presumably others, pick whatever they want for the border if it is not specified, so no border can
1381 become a visible border.

1383 */
1384 if (style->stroke.noneSet || style->stroke_width.computed == 0.0) {
1385 destroy_pen(); //this sets the NULL_PEN
1386 }

1389 Dashes converted to line segments will "open" a closed path.

1394 if (pit.end_default() != pit.end_closed()) {
1395 all_closed = false;
1396 }
1397 }
1398 }
1399 if (
1400 (style->stroke.isNone() || style->stroke.noneSet || style->stroke_width.computed == 0.0) ||
1401 (!style->stroke_dasharray.values.empty() && FixPPTDashLine) ||
1402 !all_closed
1403 ) {
1404 print_pathv(pathv, fill_transform); // do any fills. side effect: clears fill_pathv
1405 use_fill = false;
1406 }
1407 }
1408
1409 return 0;
1410}

1413unsigned int PrintEmf::stroke(
1414 Inkscape::Extension::Print * /*mod*/,
1415 Geom::PathVector const &pathv, const Geom::Affine &/*transform*/, const SPStyle *style,
1416 Geom::OptRect const &/*pbox*/, Geom::OptRect const &/*dbox*/, Geom::OptRect const &/*bbox*/)
1417{

1419 char *rec = nullptr;

1422
1423 use_stroke = true;
1424 // use_fill was set in ::fill, if it is needed
1425
1426 if (create_pen(style, tf)) {
1427 return 0;
1428 }
1429
1430 if (!style->stroke_dasharray.values.empty() && FixPPTDashLine) {
1431 // convert the path, gets its complete length, and then make a new path with parameter length instead of t
1432 Geom::Piecewise<Geom::D2<Geom::SBasis> > tmp_pathpw; // pathv-> sbasis
1433 Geom::Piecewise<Geom::D2<Geom::SBasis> > tmp_pathpw2; // sbasis using arc length parameter
1434 Geom::Piecewise<Geom::D2<Geom::SBasis> > tmp_pathpw3; // new (discontinuous) path, composed of dots/dashes
1435 Geom::Piecewise<Geom::D2<Geom::SBasis> > first_frag; // first fragment, will be appended at end
1436 int n_dash = style->stroke_dasharray.values.size();

1439 double slength = 0.0; // start of gragment
1440 double elength; // end of gragment
1441 for (const auto & i : pathv) {
1442 tmp_pathpw.concat(i.toPwSb());
1443 }
1444 tlength = length(tmp_pathpw, 0.1);
1445 tmp_pathpw2 = arc_length_parametrization(tmp_pathpw);
1446
1447 // go around the dash array repeatedly until the entire path is consumed (but not beyond).
1448 while (slength < tlength) {
1449 elength = slength + style->stroke_dasharray.values[i++].value;
1450 if (elength > tlength) {
1451 elength = tlength;
1452 }
1453 Geom::Piecewise<Geom::D2<Geom::SBasis> > fragment(portion(tmp_pathpw2, slength, elength));
1454 if (slength) {
1455 tmp_pathpw3.concat(fragment);
1456 } else {
1457 first_frag = fragment;
1458 }
