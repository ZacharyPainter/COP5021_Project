196void Inkscape::ObjectSnapper::_snapNodes(IntermSnapResults &isr,
197 SnapCandidatePoint const &p,
198 std::vector<SnapCandidatePoint> *unselected_nodes,
199 SnapConstraint const &c,
200 Geom::Point const &p_proj_on_constraint) const
201{
202 // Iterate through all nodes, find out which one is the closest to p, and snap to it!
203
204 _collectNodes(p.getSourceType(), p.getSourceNum() <= 0);
205
206 if (unselected_nodes != nullptr && unselected_nodes->size() > 0) {
207 g_assert(_points_to_snap_to != nullptr)do { (void) 0; } while (0);
208 _points_to_snap_to->insert(_points_to_snap_to->end(), unselected_nodes->begin(), unselected_nodes->end());
209 }
210
211 SnappedPoint s;
212 bool success = false;
213 bool strict_snapping = _snapmanager->snapprefs.getStrictSnapping();
214
215 for (const auto & k : *_points_to_snap_to) {
216 if (_allowSourceToSnapToTarget(p.getSourceType(), k.getTargetType(), strict_snapping)) {
217 Geom::Point target_pt = k.getPoint();
218 Geom::Coord dist = Geom::L2(target_pt - p.getPoint()); // Default: free (unconstrained) snapping
219 if (!c.isUndefined()) {
220 // We're snapping to nodes along a constraint only, so find out if this node
221 // is at the constraint, while allowing for a small margin
222 if (Geom::L2(target_pt - c.projection(target_pt)) > 1e-9) {
223 // The distance from the target point to its projection on the constraint
224 // is too large, so this point is not on the constraint. Skip it!
225 continue;
226 }
227 dist = Geom::L2(target_pt - p_proj_on_constraint);
228 }
229
230 if (dist < getSnapperTolerance() && dist < s.getSnapDistance()) {
231 s = SnappedPoint(target_pt, p.getSourceType(), p.getSourceNum(), k.getTargetType(), dist, getSnapperTolerance(), getSnapperAlwaysSnap(), false, true, k.getTargetBBox());
232 success = true;
233 }
234 }
235 }
236
237 if (success) {
238 isr.points.push_back(s);
239 }

242void Inkscape::ObjectSnapper::_snapTranslatingGuide(IntermSnapResults &isr,
243 Geom::Point const &p,
244 Geom::Point const &guide_normal) const
245{
246 // Iterate through all nodes, find out which one is the closest to this guide, and snap to it!
247 _collectNodes(SNAPSOURCE_GUIDE, true);
248
249 if (_snapmanager->snapprefs.isTargetSnappable(SNAPTARGET_PATH, SNAPTARGET_PATH_INTERSECTION, SNAPTARGET_BBOX_EDGE, SNAPTARGET_PAGE_BORDER, SNAPTARGET_TEXT_BASELINE)) {
250 _collectPaths(p, SNAPSOURCE_GUIDE, true);
251 _snapPaths(isr, SnapCandidatePoint(p, SNAPSOURCE_GUIDE), nullptr, nullptr);
252 }
253
254 SnappedPoint s;
255
256 Geom::Coord tol = getSnapperTolerance();
257
258 for (const auto & k : *_points_to_snap_to) {
259 Geom::Point target_pt = k.getPoint();
260 // Project each node (*k) on the guide line (running through point p)
261 Geom::Point p_proj = Geom::projection(target_pt, Geom::Line(p, p + Geom::rot90(guide_normal)));
262 Geom::Coord dist = Geom::L2(target_pt - p_proj); // distance from node to the guide
263 Geom::Coord dist2 = Geom::L2(p - p_proj); // distance from projection of node on the guide, to the mouse location
264 if ((dist < tol && dist2 < tol) || getSnapperAlwaysSnap()) {
265 s = SnappedPoint(target_pt, SNAPSOURCE_GUIDE, 0, k.getTargetType(), dist, tol, getSnapperAlwaysSnap(), false, true, k.getTargetBBox());
266 isr.points.push_back(s);
267 }
268 }

805void Inkscape::ObjectSnapper::_snapPathsTangPerp(bool snap_tang, bool snap_perp, IntermSnapResults &isr, SnapCandidatePoint const &p, Geom::Curve const *curve, SPDesktop const *dt) const
806{
807 // Here we will try to snap either tangentially or perpendicularly to a single path; for this we need to know where the origin is located of the line that is currently being rotated,
808 // or we need to know the vector of the guide which is currently being translated
809 std::vector<std::pair<Geom::Point, bool> > const origins_and_vectors = p.getOriginsAndVectors();
810 // Now we will iterate over all the origins and vectors and see which of these will get use a tangential or perpendicular snap
811 for (const auto & origins_and_vector : origins_and_vectors) {
812 Geom::Point origin_or_vector_doc = dt->dt2doc(origins_and_vector.first); // "first" contains a Geom::Point, denoting either a point or vector
813 if (origins_and_vector.second) { // if "second" is true then "first" is a vector, otherwise it's a point
814 // So we have a vector, which tells us what tangential or perpendicular direction we're looking for
815 if (curve->degreesOfFreedom() <= 2) { // A LineSegment has order one, and therefore 2 DOF
816 // When snapping to a point of a line segment that has a specific tangential or normal vector, then either all point
817 // along that line will be snapped to or no points at all will be snapped to. This is not very useful, so let's skip
818 // any line segments and lets only snap to higher order curves
819 continue;
820 }
821 // The vector is being treated as a point (relative to the origin), and has been translated to document coordinates accordingly
822 // We need however to make it a vector again, because also the origin has been transformed
823 origin_or_vector_doc -= dt->dt2doc(Geom::Point(0,0));
824 }
825
826 Geom::Point point_dt;
827 Geom::Coord dist;
828 std::vector<double> ts;
829
830 if (snap_tang) { // Find all points that lead to a tangential snap
831 if (origins_and_vector.second) { // if "second" is true then "first" is a vector, otherwise it's a point
832 ts = find_tangents_by_vector(origin_or_vector_doc, curve->toSBasis());
833 } else {
834 ts = find_tangents(origin_or_vector_doc, curve->toSBasis());
835 }
836 for (double t : ts) {
837 point_dt = dt->doc2dt(curve->pointAt(t));
838 dist = Geom::distance(point_dt, p.getPoint());
839 isr.points.emplace_back(point_dt, p.getSourceType(), p.getSourceNum(), SNAPTARGET_PATH_TANGENTIAL, dist, getSnapperTolerance(), getSnapperAlwaysSnap(), false, true);
840 }
841 }
842
843 if (snap_perp) { // Find all points that lead to a perpendicular snap
844 if (origins_and_vector.second) {
845 ts = find_normals_by_vector(origin_or_vector_doc, curve->toSBasis());
846 } else {
847 ts = find_normals(origin_or_vector_doc, curve->toSBasis());
848 }
849 for (double t : ts) {
850 point_dt = dt->doc2dt(curve->pointAt(t));
851 dist = Geom::distance(point_dt, p.getPoint());
852 isr.points.emplace_back(point_dt, p.getSourceType(), p.getSourceNum(), SNAPTARGET_PATH_PERPENDICULAR, dist, getSnapperTolerance(), getSnapperAlwaysSnap(), false, true);
853 }
854 }
855 }
