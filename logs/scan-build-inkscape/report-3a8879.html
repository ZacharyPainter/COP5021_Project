442void
443LPECopyRotate::doBeforeEffect (SPLPEItem const* lpeitem)
444{
445 using namespace Geom;
446 if (!split_items && lpesatellites.data().size()) {
447 processObjects(LPE_ERASE);
448 }
449 if (link_styles) {
450 reset = true;
451 }
452 if (split_items && !lpesatellites.data().size()) {

454 if (lpesatellites.data().size()) {
455 lpesatellites.update_satellites();
456 }

458 original_bbox(lpeitem, false, true);
459 if (copies_to_360 && num_copies > 2) {
460 rotation_angle.param_set_value(360.0/(double)num_copies);
461 }
462 if (method != RM_NORMAL && rotation_angle * num_copies > 360 && rotation_angle > 0 && copies_to_360) {
463 num_copies.param_set_value(floor(360/rotation_angle));
464 }
465 if (method != RM_NORMAL && mirror_copies && copies_to_360) {
466 num_copies.param_set_increments(2.0,10.0);
467 if ((int)num_copies%2 !=0) {
468 num_copies.param_set_value(num_copies+1);
469 rotation_angle.param_set_value(360.0/(double)num_copies);
470 }
471 } else {
472 num_copies.param_set_increments(1.0, 10.0);
473 }
474
475 A = Point(boundingbox_X.min(), boundingbox_Y.middle());
476 B = Point(boundingbox_X.middle(), boundingbox_Y.middle());
477 if (Geom::are_near(A, B, 0.01)) {

479 }

483 bool near_start_point = Geom::are_near(previous_start_point, (Geom::Point)starting_point, 0.01);
484 bool near_origin = Geom::are_near(previous_origin, (Geom::Point)origin, 0.01);
485 if (!near_start_point && !is_load) {
486 if (lpeitem->document->isSensitive()) {

488 }
489 if (GDK_SHIFT_MASK) {
490 dist_angle_handle = L2(B - A);
491 } else {
492 dist_angle_handle = L2(starting_point - origin);
493 }
494 }

496 dist_angle_handle = 1.0;

499 if (previous_start_point != Geom::Point(0,0) || previous_origin != Geom::Point(0,0)) {
500 distance = Geom::distance(previous_origin, starting_point);
501 }
502 start_pos = origin + dir * Rotate(-rad_from_deg(starting_angle)) * distance;
503 if (!near_start_point || !near_origin || split_items) {
504 starting_point.param_setValue(start_pos);
505 }
506
507 previous_origin = (Geom::Point)origin;
508 previous_start_point = (Geom::Point)starting_point;

511void
512LPECopyRotate::split(Geom::PathVector &path_on, Geom::Path const &divider)
513{
514 Geom::PathVector tmp_path;
515 double time_start = 0.0;
516 Geom::Path original = path_on[0];
517 int position = 0;
518 Geom::Crossings cs = crossings(original,divider);
519 std::vector<double> crossed;
520 for(auto & c : cs) {
521 crossed.push_back(c.ta);
522 }
523 std::sort(crossed.begin(), crossed.end());
524 for (double time_end : crossed) {
525 if (time_start == time_end || time_end - time_start < Geom::EPSILON) {
526 continue;
527 }
528 Geom::Path portion_original = original.portion(time_start,time_end);
529 if (!portion_original.empty()) {
530 Geom::Point side_checker = portion_original.pointAt(0.0001);
531 position = Geom::sgn(Geom::cross(divider[1].finalPoint() - divider[0].finalPoint(), side_checker - divider[0].finalPoint()));
532 if (rotation_angle != 180) {
533 position = pointInTriangle(side_checker, divider.initialPoint(), divider[0].finalPoint(), divider[1].finalPoint());
534 }
535 if (position == 1) {
536 tmp_path.push_back(portion_original);
537 }
538 portion_original.clear();
539 time_start = time_end;
540 }
541 }
542 position = Geom::sgn(Geom::cross(divider[1].finalPoint() - divider[0].finalPoint(), original.finalPoint() - divider[0].finalPoint()));
543 if (rotation_angle != 180) {
544 position = pointInTriangle(original.finalPoint(), divider.initialPoint(), divider[0].finalPoint(), divider[1].finalPoint());
545 }
546 if (cs.size() > 0 && position == 1) {
547 Geom::Path portion_original = original.portion(time_start, original.size());
548 if(!portion_original.empty()){
549 if (!original.closed()) {
550 tmp_path.push_back(portion_original);
551 } else {
552 if (tmp_path.size() > 0 && tmp_path[0].size() > 0 ) {
553 portion_original.setFinal(tmp_path[0].initialPoint());
554 portion_original.append(tmp_path[0]);
555 tmp_path[0] = portion_original;
556 } else {
557 tmp_path.push_back(portion_original);
558 }
559 }
560 portion_original.clear();
561 }
562 }
563 if (cs.size()==0 && position == 1) {
564 tmp_path.push_back(original);
565 }
566 path_on = tmp_path;
567}

569Geom::PathVector
570LPECopyRotate::doEffect_path (Geom::PathVector const & path_in)
571{
572 Geom::PathVector path_out;
573 double diagonal = Geom::distance(Geom::Point(boundingbox_X.min(),boundingbox_Y.min()),Geom::Point(boundingbox_X.max(),boundingbox_Y.max()));
574 Geom::OptRect bbox = sp_lpe_item->geometricBounds();
575 size_divider = Geom::distance(origin,bbox) + (diagonal * 6);
576 Geom::Point line_start = origin + dir * Geom::Rotate(-(Geom::rad_from_deg(starting_angle))) * size_divider;
577 Geom::Point line_end = origin + dir * Geom::Rotate(-(Geom::rad_from_deg(rotation_angle + starting_angle))) * size_divider;
578 divider = Geom::Path(line_start);
579 divider.appendNew<Geom::LineSegment>((Geom::Point)origin);
580 divider.appendNew<Geom::LineSegment>(line_end);
581 divider.close();
582 half_dir = unit_vector(Geom::middle_point(line_start,line_end) - (Geom::Point)origin);
583 FillRuleBool fillrule = fill_nonZero;
584 if (current_shape->style &&
585 current_shape->style->fill_rule.set &&
586 current_shape->style->fill_rule.computed == SP_WIND_RULE_EVENODD)
587 {
588 fillrule = (FillRuleBool)fill_oddEven;
589 }
590 if (method != RM_NORMAL) {
591 if (method != RM_KALEIDOSCOPE) {
592 path_out = doEffect_path_post(path_in, fillrule);
593 } else {
594 path_out = pathv_to_linear_and_cubic_beziers(path_in);
595 }
596 if (num_copies == 0) {
597 return path_out;
598 }
599 Geom::PathVector triangle;
600 triangle.push_back(divider);
601 path_out = sp_pathvector_boolop(path_out, triangle, bool_op_inters, fillrule, fillrule, legacytest_livarotonly);
602 if ( !split_items ) {
603 path_out = doEffect_path_post(path_out, fillrule);
604 } else {
605 path_out *= Geom::Translate(half_dir * gap);
606 }
607 } else {
608 path_out = doEffect_path_post(path_in, fillrule);
609 }
610 if (!split_items && method != RM_NORMAL) {
611 Geom::PathVector path_out_tmp;
612 for (const auto & path_it : path_out) {
613 if (path_it.empty()) {
614 continue;
615 }
616 Geom::Path::const_iterator curve_it1 = path_it.begin();
617 Geom::Path::const_iterator curve_endit = path_it.end_default();
618 Geom::Path res;
619 if (path_it.closed()) {
620 const Geom::Curve &closingline = path_it.back_closed();
621 // the closing line segment is always of type
622 // Geom::LineSegment.
623 if (are_near(closingline.initialPoint(), closingline.finalPoint())) {
624 // closingline.isDegenerate() did not work, because it only checks for
625 // *exact* zero length, which goes wrong for relative coordinates and
626 // rounding errors...
627 // the closing line segment has zero-length. So stop before that one!
628 curve_endit = path_it.end_open();
629 }
630 }
631 while (curve_it1 != curve_endit) {
632 if (!Geom::are_near(curve_it1->initialPoint(), curve_it1->pointAt(0.5), 0.05)) {
633 if (!res.empty()) {
634 res.setFinal(curve_it1->initialPoint());
635 }
636 Geom::Curve *c = curve_it1->duplicate();
637 res.append(c);
638 }
639 ++curve_it1;
640 }
641 if (path_it.closed()) {
642 res.close();
643 }
644 path_out_tmp.push_back(res);
645 }
646 path_out = path_out_tmp;
647 }
648 return pathv_to_linear_and_cubic_beziers(path_out);
649}

651Geom::PathVector
652LPECopyRotate::doEffect_path_post (Geom::PathVector const & path_in, FillRuleBool fillrule)
653{
654 if ((split_items || num_copies == 1) && method == RM_NORMAL) {
655 if (split_items) {
656 Geom::PathVector path_out = pathv_to_linear_and_cubic_beziers(path_in);
657 Geom::Affine m = Geom::Translate(-origin) * Geom::Rotate(-(Geom::rad_from_deg(starting_angle)));
658 Geom::Affine t = m * Geom::Rotate(-Geom::rad_from_deg(starting_angle)) *
659 Geom::Rotate(Geom::rad_from_deg(starting_angle)) * Geom::Translate(origin);
660 return path_out * t;
661 }
662 return path_in;
663 }
664
665 Geom::Affine pre = Geom::Translate(-origin) * Geom::Rotate(-Geom::rad_from_deg(starting_angle));
666 Geom::PathVector original_pathv = pathv_to_linear_and_cubic_beziers(path_in);
667 Geom::PathVector output_pv;
668 Geom::PathVector output;
669 for (int i = 0; i < num_copies; ++i) {
670 Geom::Rotate rot(-Geom::rad_from_deg(rotation_angle * i));
671 Geom::Affine r = Geom::identity();
672 if( i%2 != 0 && mirror_copies) {
673 r *= Geom::Rotate(Geom::Angle(half_dir)).inverse();
674 r *= Geom::Scale(1, -1);
675 r *= Geom::Rotate(Geom::Angle(half_dir));
676 }
677 Geom::Affine t = pre * r * rot * Geom::Rotate(Geom::rad_from_deg(starting_angle)) * Geom::Translate(origin);
678 if(mirror_copies && i%2 != 0) {
679 t = pre * r * rot * Geom::Rotate(Geom::rad_from_deg(starting_angle)).inverse() * Geom::Translate(origin);
680 }
681 if (method != RM_NORMAL) {
682 //we use safest way to union
683 Geom::PathVector join_pv = original_pathv * t;
684 join_pv *= Geom::Translate(half_dir * rot * gap);
685 if (!output_pv.empty()) {
686 output_pv = sp_pathvector_boolop(output_pv, join_pv, bool_op_union, fillrule, fillrule, legacytest_livarotonly);
687 } else {
688 output_pv = join_pv;
689 }
690 } else {
691 t = pre * Geom::Rotate(-Geom::rad_from_deg(starting_angle)) * r * rot * Geom::Rotate(Geom::rad_from_deg(starting_angle)) * Geom::Translate(origin);
692 if(mirror_copies && i%2 != 0) {
693 t = pre * Geom::Rotate(Geom::rad_from_deg(-starting_angle-rotation_angle)) * r * rot * Geom::Rotate(-Geom::rad_from_deg(starting_angle)) * Geom::Translate(origin);
694 }
695 output_pv = path_in * t;
696 output.insert(output.end(), output_pv.begin(), output_pv.end());
697 }
698 }
699 if (method != RM_NORMAL) {
700 output = output_pv;
701 }
702 return output;
703}
