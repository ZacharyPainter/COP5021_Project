405Geom::Affine SPText::set_transform(Geom::Affine const &xform) {
406 // See if 'shape-inside' has rectangle
407 Inkscape::Preferences *prefs = Inkscape::Preferences::get();
408 if (prefs->getBool("/tools/text/use_svg2", true)) {
409 if (style->shape_inside.set) {
410 return xform;
411 }
412 }
413 // we cannot optimize textpath because changing its fontsize will break its match to the path
414
415 if (SP_IS_TEXT_TEXTPATH (this)(SP_IS_TEXT(this) && this->firstChild() &&
SP_IS_TEXTPATH(this->firstChild()))) {
416 if (!this->_optimizeTextpathText) {
417 return xform;
418 } else {
419 this->_optimizeTextpathText = false;
420 }
421 }
422
423 // we cannot optimize text with textLength because it may show different size than specified
424 if (this->attributes.getTextLength()->_set)
425 return xform;
426
427 if (this->style && this->style->inline_size.set)
428 return xform;
429
430 /* This function takes care of scaling & translation only, we return whatever parts we can't
431 handle. */
432
433// TODO: pjrm tried to use fontsize_expansion(xform) here and it works for text in that font size
434// is scaled more intuitively when scaling non-uniformly; however this necessitated using
435// fontsize_expansion instead of expansion in other places too, where it was not appropriate
436// (e.g. it broke stroke width on copy/pasting of style from horizontally stretched to vertically
437// stretched shape). Using fontsize_expansion only here broke setting the style via font
438// dialog. This needs to be investigated further.
439 double const ex = xform.descrim();
440 if (ex == 0) {
441 return xform;
442 }
443
444 Geom::Affine ret(Geom::Affine(xform).withoutTranslation());
445 ret[0] /= ex;
446 ret[1] /= ex;
447 ret[2] /= ex;
448 ret[3] /= ex;
449
450 // Adjust x/y, dx/dy
451 this->_adjustCoordsRecursive (this, xform * ret.inverse(), ex);
452
453 // Adjust font size
454 this->_adjustFontsizeRecursive (this, ex);
455
456 // Adjust stroke width
457 this->adjust_stroke_width_recursive (ex);
458
459 // Adjust pattern fill
460 this->adjust_pattern(xform * ret.inverse());
461
462 // Adjust gradient fill
463 this->adjust_gradient(xform * ret.inverse());
464
465 return ret;
466}

617unsigned SPText::_buildLayoutInput(SPObject *object, Inkscape::Text::Layout::OptionalTextTagAttrs const &parent_optional_attrs, unsigned parent_attrs_offset, bool in_textpath)
618{
619 unsigned length = 0;
620 unsigned child_attrs_offset = 0;
621 Inkscape::Text::Layout::OptionalTextTagAttrs optional_attrs;
622
623 // Per SVG spec, an object with 'display:none' doesn't contribute to text layout.
624 if (object->style->display.computed == SP_CSS_DISPLAY_NONE) {
625 return 0;
626 }
627
628 SPText* text_object = dynamic_cast<SPText*>(object);
629 SPTSpan* tspan_object = dynamic_cast<SPTSpan*>(object);
630 SPTRef* tref_object = dynamic_cast<SPTRef*>(object);
631 SPTextPath* textpath_object = dynamic_cast<SPTextPath*>(object);
632
633 if (text_object) {
634
635 bool use_xy = true;
636 bool use_dxdyrotate = true;
637
638 // SVG 2 Text wrapping.
639 if (layout.wrap_mode == Inkscape::Text::Layout::WRAP_SHAPE_INSIDE ||
640 layout.wrap_mode == Inkscape::Text::Layout::WRAP_INLINE_SIZE) {
641 use_xy = false;
642 use_dxdyrotate = false;
643 }
644
645 text_object->attributes.mergeInto(&optional_attrs, parent_optional_attrs, parent_attrs_offset, use_xy, use_dxdyrotate);
646
647 // SVG 2 Text wrapping
648 if (layout.wrap_mode == Inkscape::Text::Layout::WRAP_INLINE_SIZE) {
649
650 // For horizontal text:
651 // 'x' is used to calculate the left/right edges of the rectangle but is not
652 // needed later. If not deleted here, it will cause an incorrect positioning
653 // of the first line.
654 // 'y' is used to determine where the first line box is located and is needed
655 // during the output stage.
656 // For vertical text:
657 // Follow above but exchange 'x' and 'y'.
658 // The SVG 2 spec currently says use the 'x' and 'y' from the <text> element,
659 // if not defined in the <text> element, use the 'x' and 'y' from the first child.
660 // We only look at the <text> element. (Doing otherwise means tracking if
661 // we've found 'x' and 'y' and then creating the Shape at the end.)
662 if (is_horizontal()) {
663 // Horizontal text
664 SVGLength* y = _getFirstYLength();
665 if (y) {
666 optional_attrs.y.push_back(*y);
667 } else {
668 std::cerr << "SPText::_buildLayoutInput: No 'y' attribute value with horizontal 'inline-size'!" << std::endl;
669 }
670 } else {
671 // Vertical text
672 SVGLength* x = _getFirstXLength();
673 if (x) {
674 optional_attrs.x.push_back(*x);
675 } else {
676 std::cerr << "SPText::_buildLayoutInput: No 'x' attribute value with vertical 'inline-size'!" << std::endl;
677 }
678 }
679 }
680
681 // set textLength on the entire layout, see note in TNG-Layout.h
682 if (text_object->attributes.getTextLength()->_set) {
683 layout.textLength._set = true;
684 layout.textLength.value = text_object->attributes.getTextLength()->value;
685 layout.textLength.computed = text_object->attributes.getTextLength()->computed;
686 layout.textLength.unit = text_object->attributes.getTextLength()->unit;
687 layout.lengthAdjust = (Inkscape::Text::Layout::LengthAdjust) text_object->attributes.getLengthAdjust();
688 }
689 }
690
691 else if (tspan_object) {
692
693 // x, y attributes are stripped from some tspans marked with role="line" as we do our own line layout.
694 // This should be checked carefully, as it can undo line layout in imported SVG files.
695 bool use_xy = !in_textpath &&
696 (tspan_object->role == SP_TSPAN_ROLE_UNSPECIFIED || !tspan_object->attributes.singleXYCoordinates());
697 bool use_dxdyrotate = true;
698
699 // SVG 2 Text wrapping: see comment above.
700 if (layout.wrap_mode == Inkscape::Text::Layout::WRAP_SHAPE_INSIDE ||
701 layout.wrap_mode == Inkscape::Text::Layout::WRAP_INLINE_SIZE) {
702 use_xy = false;
703 use_dxdyrotate = false;
704 }
705
706 tspan_object->attributes.mergeInto(&optional_attrs, parent_optional_attrs, parent_attrs_offset, use_xy, use_dxdyrotate);
707
708 if (tspan_object->role != SP_TSPAN_ROLE_UNSPECIFIED) {
709 // We are doing line wrapping using sodipodi:role="line". New lines have been stripped.
710
711 // Insert paragraph break before text if not first tspan.
712 SPObject *prev_object = object->getPrev();
713 if (prev_object && dynamic_cast<SPTSpan*>(prev_object)) {
714 if (!layout.inputExists()) {
715 // Add an object to store style, needed even if there is no text. When does this happen?
716 layout.appendText("", prev_object->style, prev_object, &optional_attrs);
717 }
718 layout.appendControlCode(Inkscape::Text::Layout::PARAGRAPH_BREAK, prev_object);
719 }
720
721 // Create empty span to store info (any non-empty tspan with sodipodi:role="line" has a child).
722 if (!object->hasChildren()) {
723 layout.appendText("", object->style, object, &optional_attrs);
724 }
725
726 length++; // interpreting line breaks as a character for the purposes of x/y/etc attributes
727 // is a liberal interpretation of the svg spec, but a strict reading would mean
728 // that if the first line is empty the second line would take its place at the
729 // start position. Very confusing.
730 // SVG 2 clarifies, attributes are matched to unicode input characters so line
731 // breaks do match to an x/y/etc attribute.
732 child_attrs_offset--;
733 }
734 }
735
736 else if (tref_object) {
737 tref_object->attributes.mergeInto(&optional_attrs, parent_optional_attrs, parent_attrs_offset, true, true);
738 }
739
740 else if (textpath_object) {
741 in_textpath = true; // This should be made local so we can mix normal text with textpath per SVG 2.
742 textpath_object->attributes.mergeInto(&optional_attrs, parent_optional_attrs, parent_attrs_offset, false, true);
743 optional_attrs.x.clear(); // Hmm, you can use x with horizontal text. So this is probably wrong.
744 optional_attrs.y.clear();
745 }
746
747 else {
748 optional_attrs = parent_optional_attrs;
749 child_attrs_offset = parent_attrs_offset;
750 }
751
752 // Recurse
753 for (auto& child: object->children) {
754 SPString *str = dynamic_cast<SPString *>(&child);
755 if (str) {
756 Glib::ustring const &string = str->string;
757 // std::cout << " Appending: >" << string << "<" << std::endl;
758 layout.appendText(string, object->style, &child, &optional_attrs, child_attrs_offset + length);
759 length += string.length();
760 } else if (!sp_repr_is_meta_element(child.getRepr())) {
761 /* ^^^^ XML Tree being directly used here while it shouldn't be.*/
762 length += _buildLayoutInput(&child, optional_attrs, child_attrs_offset + length, in_textpath);
763 }
764 }
765
766 return length;
767}

1617void TextTagAttributes::transform(Geom::Affine const &matrix, double scale_x, double scale_y, bool extend_zero_length)
1618{
1619 SVGLength zero_length;
1620 zero_length = 0.0;
1621
1622 /* edge testcases for this code:
1623 1) moving text elements whose position is done entirely with transform="...", no x,y attributes
1624 2) unflowing multi-line flowtext then moving it (it has x but not y)
1625 */
1626 unsigned points_count = std::max(attributes.x.size(), attributes.y.size());
1627 if (extend_zero_length && points_count < 1)
1628 points_count = 1;
1629 for (unsigned i = 0 ; i < points_count ; i++) {
1630 Geom::Point point;
1631 if (i < attributes.x.size()) point[Geom::X] = attributes.x[i].computed;
1632 else point[Geom::X] = 0.0;
1633 if (i < attributes.y.size()) point[Geom::Y] = attributes.y[i].computed;
1634 else point[Geom::Y] = 0.0;
1635 point *= matrix;
1636 if (i < attributes.x.size())
1637 attributes.x[i] = point[Geom::X];
1638 else if (point[Geom::X] != 0.0 && extend_zero_length) {
1639 attributes.x.resize(i + 1, zero_length);
1640 attributes.x[i] = point[Geom::X];
1641 }
1642 if (i < attributes.y.size())
1643 attributes.y[i] = point[Geom::Y];
1644 else if (point[Geom::Y] != 0.0 && extend_zero_length) {
1645 attributes.y.resize(i + 1, zero_length);
1646 attributes.y[i] = point[Geom::Y];
1647 }
1648 }
1649 for (auto & it : attributes.dx)
1650 it = it.computed * scale_x;
1651 for (auto & it : attributes.dy)
1652 it = it.computed * scale_y;
1653}
