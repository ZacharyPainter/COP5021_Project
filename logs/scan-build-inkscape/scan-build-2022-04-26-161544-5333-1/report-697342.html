22static Point intersection_point(Point origin_a, Point vector_a, Point origin_b, Point vector_b)
23{
24 Coord denom = cross(vector_a, vector_b);
25 if (!are_near(denom,0.)) {
26 Coord t = (cross(vector_b, origin_a) + cross(origin_b, vector_b)) / denom;
27 return origin_a + vector_a*t;
28 }
29 return Point(infinity(), infinity());
30}

32/**
33* Find circle that touches inside of the curve, with radius matching the curvature, at time value \c t.
34* Because this method internally uses unitTangentAt, t should be smaller than 1.0 (see unitTangentAt).
35*/
36static Circle touching_circle( D2<SBasis> const &curve, double t, double tol=0.01 )
37{
38 D2<SBasis> dM=derivative(curve);
39 if ( are_near(L2sq(dM(t)), tol) ) {
40 dM=derivative(dM);
41 }
42 if ( are_near(L2sq(dM(t)), tol) ) { // try second time
43 dM=derivative(dM);
44 }
45 Piecewise<D2<SBasis> > unitv = unitVector(dM,tol);
46 Piecewise<SBasis> dMlength = dot(Piecewise<D2<SBasis> >(dM),unitv);
47 Piecewise<SBasis> k = cross(derivative(unitv),unitv);
48 k = divide(k,dMlength,tol,3);
49 double curv = k(t); // note that this value is signed
50
51 Geom::Point normal = unitTangentAt(curve, t).cw();
52 double radius = 1/curv;
53 Geom::Point center = curve(t) + radius*normal;
54 return Geom::Circle(center, fabs(radius));
55}

63 return( 0.5 * fabs( ( a[X]*(b[Y]-c[Y]) + b[X]*(c[Y]-a[Y]) + c[X]*(a[Y]-b[Y]) ) ) );

67static Circle touching_circle( CubicBezier const &curve, bool start ) {

69 double k = 0;
70 Geom::Point p;
71 Geom::Point normal;
72 if ( start ) {
73 double distance = Geom::distance( curve[1], curve[0] );
74 k = 4.0/3.0 * area( curve[0], curve[1], curve[2] ) /

77 k = -k;
78 }
79 p = curve[0];
80 normal = Geom::Point(curve[1] - curve[0]).cw();
81 normal.normalize();

83 } else {
84 double distance = Geom::distance( curve[3], curve[2] );
85 k = 4.0/3.0 * area( curve[1], curve[2], curve[3] ) /
86 (distance * distance * distance);
87 if( Geom::cross(curve[1]-curve[2], curve[2]-curve[3]) < 0 ) {
88 k = -k;
89 }

145void miter_join_internal(join_data const &jd, bool clip)
146{
147 using namespace Geom;
148
149 Curve const& incoming = jd.res.back();
150 Curve const& outgoing = jd.outgoing.front();
151 Path &res = jd.res;
152 double width = jd.width, miter = jd.miter;
153
154 Point tang1 = jd.in_tang;
155 Point tang2 = jd.out_tang;
156 Point p = intersection_point(incoming.finalPoint(), tang1, outgoing.initialPoint(), tang2);
157
158 bool satisfied = false;
159 bool inc_ls = res.back_open().degreesOfFreedom() <= 4;
160
161 if (p.isFinite()) {
162 // check size of miter
163 Point point_on_path = incoming.finalPoint() + rot90(tang1)*width;
164 // SVG defines miter length as distance between inner intersection and outer intersection,
165 // which is twice the distance from p to point_on_path but width is half stroke width.
166 satisfied = distance(p, point_on_path) <= miter * width;
167 if (satisfied) {
168 // miter OK, check to see if we can do a relocation
169 if (inc_ls) {
170 res.setFinal(p);
171 } else {
172 res.appendNew<LineSegment>(p);
173 }
174 } else if (clip) {
175 // std::cout << " Clipping ------------ " << std::endl;
176 // miter needs clipping, find two points
177 Point bisector_versor = Line(point_on_path, p).versor();
178 Point point_limit = point_on_path + miter * width * bisector_versor;
179 // std::cout << " bisector_versor: " << bisector_versor << std::endl;
180 // std::cout << " point_limit: " << point_limit << std::endl;
181 Point p1 = intersection_point(incoming.finalPoint(), tang1, point_limit, bisector_versor.cw());
182 Point p2 = intersection_point(outgoing.initialPoint(), tang2, point_limit, bisector_versor.cw());
183 // std::cout << " p1: " << p1 << std::endl;
184 // std::cout << " p2: " << p2 << std::endl;
185 if (inc_ls) {
186 res.setFinal(p1);
187 } else {
188 res.appendNew<LineSegment>(p1);
189 }
190 res.appendNew<LineSegment>(p2);
191 }
192 }
193
194 res.appendNew<LineSegment>(outgoing.initialPoint());
195
196 // check if we can do another relocation
197 bool out_ls = outgoing.degreesOfFreedom() <= 4;
198
199 if ((satisfied || clip) && out_ls) {
200 res.setFinal(outgoing.finalPoint());
201 } else {
202 res.append(outgoing);
203 }
204
205 // either way, add the rest of the path
206 res.insert(res.end(), ++jd.outgoing.begin(), jd.outgoing.end());
207}

212Geom::Point pick_solution(std::vector<Geom::ShapeIntersection> points, Geom::Point tang2, Geom::Point endPt)
213{
214 assert(points.size() == 2)(static_cast<void> (0));
215 Geom::Point sol;
216 if ( dot(tang2, points[0].point() - endPt) > 0 ) {
217 // points[0] is bad, choose points[1]
218 sol = points[1];
219 } else if ( dot(tang2, points[1].point() - endPt) > 0 ) { // points[0] could be good, now check points[1]
220 // points[1] is bad, choose points[0]
221 sol = points[0];
222 } else {
223 // both points are good, choose nearest
224 sol = ( distanceSq(endPt, points[0].point()) < distanceSq(endPt, points[1].point()) )

226 }

981Geom::PathVector outline(
982 Geom::Path const& input,
983 double width,
984 double miter,
985 LineJoinType join,
986 LineCapType butt,
987 double tolerance)
988{
989 if (input.size() == 0) return Geom::PathVector(); // nope, don't even try
990
991 Geom::PathBuilder res;
992 Geom::Path with_dir = half_outline(input, width/2., miter, join, tolerance);
993 Geom::Path against_dir = half_outline(input.reversed(), width/2., miter, join, tolerance);
994 res.moveTo(with_dir[0].initialPoint());
995 res.append(with_dir);
996
997 cap_func *cf;
998 switch (butt) {
999 case BUTT_ROUND:
1000 cf = &round_cap;
1001 break;
1002 case BUTT_SQUARE:
1003 cf = &square_cap;
1004 break;
1005 case BUTT_PEAK:
1006 cf = &peak_cap;
1007 break;
1008 default:
1009 cf = &flat_cap;
1010 }
1011
1012 // glue caps
1013 if (!input.closed()) {
1014 cf(res, with_dir, against_dir, width);
1015 } else {
1016 res.closePath();
1017 res.moveTo(against_dir.initialPoint());
1018 }
1019
1020 res.append(against_dir);
1021
1022 if (!input.closed()) {
1023 cf(res, against_dir, with_dir, width);
1024 }
1025
1026 res.closePath();
1027 res.flush();
1028 return res.peek();
1029}
1030
1031Geom::Path half_outline(
1032 Geom::Path const& input,
1033 double width,
1034 double miter,
1035 LineJoinType join,
1036 double tolerance)
1037{
1038 if (tolerance <= 0) {
1039 if (std::abs(width) > 0) {
1040 tolerance = 5.0 * (std::abs(width)/100);
1041 }
1042 else {
1043 tolerance = 1e-4;
1044 }
1045 }
1046 Geom::Path res;
1047 if (input.size() == 0) return res;
1048
1049 Geom::Point tang1 = input[0].unitTangentAt(0);
1050 Geom::Point start = input.initialPoint() + tang1 * width;
1051 Geom::Path temp;
1052 Geom::Point tang[2];
1053
1054 res.setStitching(true);
1055 temp.setStitching(true);
1056
1057 res.start(start);
1058
1059 // Do two curves at a time for efficiency, since the join function needs to know the outgoing curve as well
1060 const Geom::Curve &closingline = input.back_closed();
1061 const size_t k = (are_near(closingline.initialPoint(), closingline.finalPoint()) && input.closed() )
1062 ?input.size_open():input.size_default();
1063
1064 for (size_t u = 0; u < k; u += 2) {
1065 temp.clear();
1066
1067 offset_curve(temp, &input[u], width, tolerance);
1068
1069 // on the first run through, there isn't a join
1070 if (u == 0) {
1071 res.append(temp);
1072 } else {
1073 tangents(tang, input[u-1], input[u]);
1074 outline_join(res, temp, tang[0], tang[1], width, miter, join);
1075 }
1076
1077 // odd number of paths
1078 if (u < k - 1) {
1079 temp.clear();
1080 offset_curve(temp, &input[u+1], width, tolerance);
1081 tangents(tang, input[u], input[u+1]);
1082 outline_join(res, temp, tang[0], tang[1], width, miter, join);
1083 }
1084 }
1085 if (input.closed()) {
1086 Geom::Curve const &c1 = res.back();
1087 Geom::Curve const &c2 = res.front();
1088 temp.clear();
1089 temp.append(c1);
1090 Geom::Path temp2;
1091 temp2.append(c2);
1092 tangents(tang, input.back(), input.front());
1093 outline_join(temp, temp2, tang[0], tang[1], width, miter, join);
1094 res.erase(res.begin());
1095 res.erase_last();
1096 res.append(temp);
1097 res.close();
1098 }
1099 return res;
1100}
1101
1102void outline_join(Geom::Path &res, Geom::Path const& temp, Geom::Point in_tang, Geom::Point out_tang, double width, double miter, Inkscape::LineJoinType join)
1103{
1104 if (res.size() == 0 || temp.size() == 0)
1105 return;
1106 Geom::Curve const& outgoing = temp.front();
1107 if (Geom::are_near(res.finalPoint(), outgoing.initialPoint(), 0.01)) {
1108 // if the points are /that/ close, just ignore this one
1109 res.setFinal(temp.initialPoint());
1110 res.append(temp);
1111 return;
1112 }
1113
1114 join_data jd(res, temp, in_tang, out_tang, miter, width);
1115 if (!(Geom::cross(in_tang, out_tang) > 0)) {
1116 join = Inkscape::JOIN_BEVEL;
1117 }
1118 join_func *jf;
1119 switch (join) {
1120 case Inkscape::JOIN_BEVEL:
1121 jf = &bevel_join;
1122 break;
1123 case Inkscape::JOIN_ROUND:
1124 jf = &round_join;
1125 break;
1126 case Inkscape::JOIN_EXTRAPOLATE:
1127 jf = &extrapolate_join;
1128 break;
1129 case Inkscape::JOIN_EXTRAPOLATE1:
1130 jf = &extrapolate_join_alt1;
1131 break;
1132 case Inkscape::JOIN_EXTRAPOLATE2:
1133 jf = &extrapolate_join_alt2;
1134 break;
1135 case Inkscape::JOIN_EXTRAPOLATE3:
1136 jf = &extrapolate_join_alt3;
1137 break;
1138 case Inkscape::JOIN_MITER_CLIP:
1139 jf = &miter_clip_join;
1140 break;
1141 default:
1142 jf = &miter_join;
1143 }
1144 jf(jd);
1145 }
