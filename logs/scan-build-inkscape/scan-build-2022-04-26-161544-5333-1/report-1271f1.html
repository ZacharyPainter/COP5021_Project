49/* The class below implement the following hack:
50 *
51 * The pixels formats of Cairo and GdkPixbuf are different.
52 * GdkPixbuf accesses pixels as bytes, alpha is not premultiplied,
53 * and successive bytes of a single pixel contain R, G, B and A components.
54 * Cairo accesses pixels as 32-bit ints, alpha is premultiplied,
55 * and each int contains as 0xAARRGGBB, accessed with bitwise operations.
56 *
57 * In other words, on a little endian system, a GdkPixbuf will contain:
58 * char *data = "rgbargbargba...."
59 * int *data = { 0xAABBGGRR, 0xAABBGGRR, 0xAABBGGRR, ... }
60 * while a Cairo image surface will contain:
61 * char *data = "bgrabgrabgra...."
62 * int *data = { 0xAARRGGBB, 0xAARRGGBB, 0xAARRGGBB, ... }
63 *
64 * It is possible to convert between these two formats (almost) losslessly.
65 * Some color information from partially transparent regions of the image
66 * is lost, but the result when displaying this image will remain the same.
67 *
68 * The class allows interoperation between GdkPixbuf
69 * and Cairo surfaces without creating a copy of the image.
70 * This is implemented by creating a GdkPixbuf and a Cairo image surface
71 * which share their data. Depending on what is needed at a given time,
72 * the pixels are converted in place to the Cairo or the GdkPixbuf format.
73 */

607/*
608 * Can be called recursively.
609 * If optimize_stroke == false, the view Rect is not used.
610 */
611static void
612feed_curve_to_cairo(cairo_t *cr, Geom::Curve const &c, Geom::Affine const & trans, Geom::Rect view, bool optimize_stroke)
613{
614 using Geom::X;
615 using Geom::Y;
616
617 unsigned order = 0;
618 if (Geom::BezierCurve const* b = dynamic_cast<Geom::BezierCurve const*>(&c)) {
619 order = b->order();
620 }
621
622 // handle the three typical curve cases
623 switch (order) {
624 case 1:
625 {
626 Geom::Point end_tr = c.finalPoint() * trans;
627 if (!optimize_stroke) {
628 cairo_line_to(cr, end_tr[0], end_tr[1]);
629 } else {
630 Geom::Rect swept(c.initialPoint()*trans, end_tr);
631 if (swept.intersects(view)) {
632 cairo_line_to(cr, end_tr[0], end_tr[1]);
633 } else {
634 cairo_move_to(cr, end_tr[0], end_tr[1]);
635 }
636 }
637 }
638 break;
639 case 2:
640 {
641 Geom::QuadraticBezier const *quadratic_bezier = static_cast<Geom::QuadraticBezier const*>(&c);
642 std::vector<Geom::Point> points = quadratic_bezier->controlPoints();
643 points[0] *= trans;
644 points[1] *= trans;
645 points[2] *= trans;
646 // degree-elevate to cubic Bezier, since Cairo doesn't do quadratic Beziers
647 Geom::Point b1 = points[0] + (2./3) * (points[1] - points[0]);
648 Geom::Point b2 = b1 + (1./3) * (points[2] - points[0]);
649 if (!optimize_stroke) {
650 cairo_curve_to(cr, b1[X], b1[Y], b2[X], b2[Y], points[2][X], points[2][Y]);
651 } else {
652 Geom::Rect swept(points[0], points[2]);
653 swept.expandTo(points[1]);
654 if (swept.intersects(view)) {
655 cairo_curve_to(cr, b1[X], b1[Y], b2[X], b2[Y], points[2][X], points[2][Y]);
656 } else {
657 cairo_move_to(cr, points[2][X], points[2][Y]);
658 }
659 }
660 }
661 break;
662 case 3:
663 {
664 Geom::CubicBezier const *cubic_bezier = static_cast<Geom::CubicBezier const*>(&c);
665 std::vector<Geom::Point> points = cubic_bezier->controlPoints();
666 //points[0] *= trans; // don't do this one here for fun: it is only needed for optimized strokes
667 points[1] *= trans;
668 points[2] *= trans;
669 points[3] *= trans;
670 if (!optimize_stroke) {
671 cairo_curve_to(cr, points[1][X], points[1][Y], points[2][X], points[2][Y], points[3][X], points[3][Y]);
672 } else {
673 points[0] *= trans; // didn't transform this point yet
674 Geom::Rect swept(points[0], points[3]);
675 swept.expandTo(points[1]);
676 swept.expandTo(points[2]);
677 if (swept.intersects(view)) {
678 cairo_curve_to(cr, points[1][X], points[1][Y], points[2][X], points[2][Y], points[3][X], points[3][Y]);
679 } else {
680 cairo_move_to(cr, points[3][X], points[3][Y]);
681 }
682 }
683 }
684 break;
685 default:
686 {
687 if (Geom::EllipticalArc const *arc = dynamic_cast<Geom::EllipticalArc const*>(&c)) {
688 if (arc->isChord()) {
689 Geom::Point endPoint(arc->finalPoint());
690 cairo_line_to(cr, endPoint[0], endPoint[1]);
691 } else {
692 Geom::Affine xform = arc->unitCircleTransform() * trans;
693 // Don't draw anything if the angle is borked
694 if(std::isnan(arc->initialAngle()) || std::isnan(arc->finalAngle())) {
695 g_warning("Bad angle while drawing EllipticalArc");
696 break;
697 }
698
699 // Apply the transformation to the current context
700 cairo_matrix_t cm;
701 cm.xx = xform[0];
702 cm.xy = xform[2];
703 cm.x0 = xform[4];
704 cm.yx = xform[1];
705 cm.yy = xform[3];
706 cm.y0 = xform[5];
707
708 cairo_save(cr);
709 cairo_transform(cr, &cm);
710
711 // Draw the circle
712 if (arc->sweep()) {
713 cairo_arc(cr, 0, 0, 1, arc->initialAngle(), arc->finalAngle());
714 } else {
715 cairo_arc_negative(cr, 0, 0, 1, arc->initialAngle(), arc->finalAngle());
716 }
717 // Revert the current context
718 cairo_restore(cr);
719 }
720 } else {
721 // handles sbasis as well as all other curve types
722 // this is very slow
723 Geom::Path sbasis_path = Geom::cubicbezierpath_from_sbasis(c.toSBasis(), 0.1);
724
725 // recurse to convert the new path resulting from the sbasis to svgd
726 for (const auto & iter : sbasis_path) {
727 feed_curve_to_cairo(cr, iter, trans, view, optimize_stroke);
728 }
729 }
730 }
731 break;
732 }
733}

736/** Feeds path-creating calls to the cairo context translating them from the Path */
737static void
738feed_path_to_cairo (cairo_t *ct, Geom::Path const &path)
739{
740 if (path.empty())
741 return;
742
743 cairo_move_to(ct, path.initialPoint()[0], path.initialPoint()[1] );
744
745 for (Geom::Path::const_iterator cit = path.begin(); cit != path.end_open(); ++cit) {
746 feed_curve_to_cairo(ct, *cit, Geom::identity(), Geom::Rect(), false); // optimize_stroke is false, so the view rect is not used
747 }
748
749 if (path.closed()) {
750 cairo_close_path(ct);
751 }
752}
753
754/** Feeds path-creating calls to the cairo context translating them from the Path, with the given transform and shift */
755static void
756feed_path_to_cairo (cairo_t *ct, Geom::Path const &path, Geom::Affine trans, Geom::OptRect area, bool optimize_stroke, double stroke_width)
757{
758 if (!area)
759 return;
760 if (path.empty())
761 return;
762
763 // Transform all coordinates to coords within "area"
764 Geom::Point shift = area->min();
765 Geom::Rect view = *area;
766 view.expandBy (stroke_width);
767 view = view * (Geom::Affine)Geom::Translate(-shift);
768 // Pass transformation to feed_curve, so that we don't need to create a whole new path.
769 Geom::Affine transshift(trans * Geom::Translate(-shift));
770
771 Geom::Point initial = path.initialPoint() * transshift;
772 cairo_move_to(ct, initial[0], initial[1] );
773
774 for(Geom::Path::const_iterator cit = path.begin(); cit != path.end_open(); ++cit) {
775 feed_curve_to_cairo(ct, *cit, transshift, view, optimize_stroke);
776 }
777
778 if (path.closed()) {
779 if (!optimize_stroke) {
780 cairo_close_path(ct);
781 } else {
782 cairo_line_to(ct, initial[0], initial[1]);
783 /* We cannot use cairo_close_path(ct) here because some parts of the path may have been
784 clipped and not drawn (maybe the before last segment was outside view area), which
785 would result in closing the "subpath" after the last interruption, not the entire path.
786
787 However, according to cairo documentation:
788 The behavior of cairo_close_path() is distinct from simply calling cairo_line_to() with the equivalent coordinate
789 in the case of stroking. When a closed sub-path is stroked, there are no caps on the ends of the sub-path. Instead,
790 there is a line join connecting the final and initial segments of the sub-path.
791
792 The correct fix will be possible when cairo introduces methods for moving without
793 ending/starting subpaths, which we will use for skipping invisible segments; then we
794 will be able to use cairo_close_path here. This issue also affects ps/eps/pdf export,
795 see bug 168129
796 */
797 }
798 }

1472/**
1473 * Draw drop shadow around the 'rect' with given 'size' and 'color'; shadow extends to the right and bottom of rect.
1474 */
1475void ink_cairo_draw_drop_shadow(const Cairo::RefPtr<Cairo::Context> &ctx, const Geom::Rect& rect, double size, guint32 color, double color_alpha) {
1476 // draw fake drop shadow built from gradients
1477 const auto r = SP_RGBA32_R_F(color)(((((color) >> 24) & 0xff)) / 255.0);
1478 const auto g = SP_RGBA32_G_F(color)(((((color) >> 16) & 0xff)) / 255.0);
1479 const auto b = SP_RGBA32_B_F(color)(((((color) >> 8) & 0xff)) / 255.0);
1480 const auto a = color_alpha;
1481 const Geom::Point corners[] = { rect.corner(0), rect.corner(1), rect.corner(2), rect.corner(3) };
1482 // space for gradient shadow
1483 double sw = size;
1484 double half = sw / 2;
1485 using Geom::X;
1486 using Geom::Y;
1487 // 8 gradients total: 4 sides + 4 corners
1488 auto grad_top = Cairo::LinearGradient::create(0, corners[0][Y] + half, 0, corners[0][Y] - half);
1489 auto grad_right = Cairo::LinearGradient::create(corners[1][X], 0, corners[1][X] + sw, 0);
1490 auto grad_bottom = Cairo::LinearGradient::create(0, corners[2][Y], 0, corners[2][Y] + sw);
1491 auto grad_left = Cairo::LinearGradient::create(corners[0][X] + half, 0, corners[0][X] - half, 0);
1492 auto grad_btm_right = Cairo::RadialGradient::create(corners[2][X], corners[2][Y], 0, corners[2][X], corners[2][Y], sw);
1493 auto grad_top_right = Cairo::RadialGradient::create(corners[1][X], corners[1][Y] + half, 0, corners[1][X], corners[1][Y] + half, sw);
1494 auto grad_btm_left = Cairo::RadialGradient::create(corners[3][X] + half, corners[3][Y], 0, corners[3][X] + half, corners[3][Y], sw);
1495 auto grad_top_left = Cairo::RadialGradient::create(corners[0][X], corners[0][Y], 0, corners[0][X], corners[0][Y], half);
1496 const int N = 15; // number of gradient stops; stops used to make it non-linear
1497 // using easing function here: (exp(a*(1-t)) - 1) / (exp(a) - 1);
1498 // it has a nice property of growing from 0 to 1 for t in [0..1]
1499 const auto A = 4.0; // this coefficient changes how steep the curve is and controls shadow drop-off
1500 const auto denominator = exp(A) - 1;
1501 for (int i = 0; i <= N; ++i) {
1502 auto pos = static_cast<double>(i) / N;
1503 // exponential decay for drop shadow - long tail, with values from 100% down to 0% opacity
1504 auto t = 1 - pos; // reverse 't' so alpha drops from 1 to 0
1505 auto alpha = (exp(A * t) - 1) / denominator;
1506 grad_top->add_color_stop_rgba(pos, r, g, b, alpha * a);
1507 grad_bottom->add_color_stop_rgba(pos, r, g, b, alpha * a);
1508 grad_right->add_color_stop_rgba(pos, r, g, b, alpha * a);
1509 grad_left->add_color_stop_rgba(pos, r, g, b, alpha * a);
1510 grad_btm_right->add_color_stop_rgba(pos, r, g, b, alpha * a);
1511 grad_top_right->add_color_stop_rgba(pos, r, g, b, alpha * a);
1512 grad_btm_left->add_color_stop_rgba(pos, r, g, b, alpha * a);
1513 // this left/top corner is just a silver of the shadow: half of it is "hidden" beneath the page
1514 if (pos >= 0.5) {
1515 grad_top_left->add_color_stop_rgba(2 * (pos - 0.5), r, g, b, alpha * a);
1516 }
1517 }
1518
1519 // shadow at the top (faint)
1520 ctx->rectangle(corners[0][X], corners[0][Y] - half, std::max(corners[1][X] - corners[0][X], 0.0), half);
1521 ctx->set_source(grad_top);
1522 ctx->fill();
1523
1524 // right side
1525 ctx->rectangle(corners[1][X], corners[1][Y] + half, sw, std::max(corners[2][Y] - corners[1][Y] - half, 0.0));
1526 ctx->set_source(grad_right);
1527 ctx->fill();
1528
1529 // bottom side
1530 ctx->rectangle(corners[0][X] + half, corners[2][Y], std::max(corners[1][X] - corners[0][X] - half, 0.0), sw);
1531 ctx->set_source(grad_bottom);
1532 ctx->fill();
1533
1534 // left side (faint)
1535 ctx->rectangle(corners[0][X] - half, corners[0][Y], half, std::max(corners[2][Y] - corners[1][Y], 0.0));
1536 ctx->set_source(grad_left);
1537 ctx->fill();
1538
1539 // bottom corners
1540 ctx->rectangle(corners[2][X], corners[2][Y], sw, sw);
1541 ctx->set_source(grad_btm_right);
1542 ctx->fill();
1543
1544 ctx->rectangle(corners[3][X] - half, corners[3][Y], std::min(sw, rect.width() + half), sw);
1545 ctx->set_source(grad_btm_left);
1546 ctx->fill();
1547
1548 // top corners
1549 ctx->rectangle(corners[1][X], corners[1][Y] - half, sw, std::min(sw, rect.height() + half));
1550 ctx->set_source(grad_top_right);
1551 ctx->fill();
1552
1553 ctx->rectangle(corners[0][X] - half, corners[0][Y] - half, half, half);
1554 ctx->set_source(grad_top_left);
1555 ctx->fill();
1556}

1596/* The following two functions use "from" instead of "to", because when you write:
1597 val1 = argb32_from_pixbuf(val1);
1598 the name of the format is closer to the value in that format. */
1599
1600guint32 argb32_from_pixbuf(guint32 c)
1601{
1602 guint32 o = 0;
1603#if G_BYTE_ORDER1234 == G_LITTLE_ENDIAN1234
1604 guint32 a = (c & 0xff000000) >> 24;
1605#else
1606 guint32 a = (c & 0x000000ff);
1607#endif
1608 if (a != 0) {
1609 // extract color components
1610#if G_BYTE_ORDER1234 == G_LITTLE_ENDIAN1234
1611 guint32 r = (c & 0x000000ff);
1612 guint32 g = (c & 0x0000ff00) >> 8;
1613 guint32 b = (c & 0x00ff0000) >> 16;
1614#else
1615 guint32 r = (c & 0xff000000) >> 24;
1616 guint32 g = (c & 0x00ff0000) >> 16;
1617 guint32 b = (c & 0x0000ff00) >> 8;
1618#endif
1619 // premultiply
1620 r = premul_alpha(r, a);
1621 b = premul_alpha(b, a);
1622 g = premul_alpha(g, a);
1623 // combine into output
1624 o = (a << 24) | (r << 16) | (g << 8) | (b);
1625 }
1626 return o;
1627}

1629/**
1630 * Convert one pixel from ARGB to GdkPixbuf format.
1631 *
1632 * @param c ARGB color
1633 * @param bgcolor Color to use if c.alpha is zero (bgcolor.alpha is ignored)
1634 */
1635guint32 pixbuf_from_argb32(guint32 c, guint32 bgcolor)
1636{
1637 guint32 a = (c & 0xff000000) >> 24;
1638 if (a == 0) {
1639 assert(c == 0)(static_cast<void> (0));
1640 c = bgcolor;
1641 }
1642
1643 // extract color components
1644 guint32 r = (c & 0x00ff0000) >> 16;
1645 guint32 g = (c & 0x0000ff00) >> 8;
1646 guint32 b = (c & 0x000000ff);
1647
1648 if (a != 0) {
1649 r = unpremul_alpha(r, a);
1650 g = unpremul_alpha(g, a);
1651 b = unpremul_alpha(b, a);
1652 }
1653
1654 // combine into output
1655#if G_BYTE_ORDER1234 == G_LITTLE_ENDIAN1234
1656 guint32 o = (r) | (g << 8) | (b << 16) | (a << 24);
1657#else
1658 guint32 o = (r << 24) | (g << 16) | (b << 8) | (a);
1659#endif
1660 return o;
1661}

1762/**
1763 * Converts a pixbuf to a PNG data structure.
1764 * For 8-but RGBA png, this is like copying.
1765 *
1766 */
1767const guchar* pixbuf_to_png(guchar const**rows, guchar* px, int num_rows, int num_cols, int stride, int color_type, int bit_depth)
1768{
1769 int n_fields = 1 + (color_type&2) + (color_type&4)/4;
1770 const guchar* new_data = (const guchar*)malloc(((n_fields * bit_depth * num_cols + 7)/8) * num_rows);
1771 char* ptr = (char*) new_data;
1772 // Used when we write image data smaller than one byte (for instance in
1773 // black and white images where 1px = 1bit). Only possible with greyscale.
1774 int pad = 0;
1775 for (int row = 0; row < num_rows; ++row) {
1776 rows[row] = (const guchar*)ptr;
1777 for (int col = 0; col < num_cols; ++col) {
1778 guint32 *pixel = reinterpret_cast<guint32*>(px + row*stride)+col;
1779
1780 guint64 pix3 = (*pixel & 0xff000000) >> 24;
1781 guint64 pix2 = (*pixel & 0x00ff0000) >> 16;
1782 guint64 pix1 = (*pixel & 0x0000ff00) >> 8;
1783 guint64 pix0 = (*pixel & 0x000000ff);
1784
1785#if G_BYTE_ORDER1234 == G_LITTLE_ENDIAN1234
1786 guint64 a = pix3, b = pix2, g = pix1, r = pix0;
1787#else
1788 guint64 r = pix3, g = pix2, b = pix1, a = pix0;
1789#endif
1790
1791 // One of possible rgb to greyscale formulas. This one is called "luminance", "luminosity" or "luma"
1792 guint16 gray = (guint16)((guint32)((0.2126*(r<<24) + 0.7152*(g<<24) + 0.0722*(b<<24)))>>16);
1793
1794 if (color_type & 2) { // RGB or RGBA
1795 // for 8bit->16bit transition, I take the FF -> FFFF convention (multiplication by 0x101).
1796 // If you prefer FF -> FF00 (multiplication by 0x100), remove the <<8, <<24, <<40 and <<56
1797 // for little-endian, and remove the <<0, <<16, <<32 and <<48 for big-endian.
1798 if (color_type & 4) { // RGBA
1799 if (bit_depth == 8)
1800 *((guint32*)ptr) = *pixel;
1801 else
1802 // This uses the samples in the order they appear in pixel rather than
1803 // normalised to abgr or rgba in order to make it endian agnostic,
1804 // exploiting the symmetry of the expression (0x101 is the same in both
1805 // endiannesses and each sample is multiplied by that).
1806 *((guint64*)ptr) = (guint64)((pix3<<56)+(pix3<<48)+(pix2<<40)+(pix2<<32)+(pix1<<24)+(pix1<<16)+(pix0<<8)+(pix0));
1807 } else { // RGB
1808 if (bit_depth == 8) {
1809 *ptr = r;
1810 *(ptr+1) = g;
1811 *(ptr+2) = b;
1812 } else {
1813 *((guint16*)ptr) = (r<<8)+r;
1814 *((guint16*)(ptr+2)) = (g<<8)+g;
1815 *((guint16*)(ptr+4)) = (b<<8)+b;
1816 }
1817 }
1818 } else { // Grayscale
1819 if (bit_depth == 16) {
1820#if G_BYTE_ORDER1234 == G_LITTLE_ENDIAN1234
1821 *(guint16*)ptr = ((gray & 0xff00)>>8) + ((gray & 0x00ff)<<8);
1822#else
1823 *(guint16*)ptr = gray;
1824#endif
1825 // For 8bit->16bit this mirrors RGB(A), multiplying by
1826 // 0x101; if you prefer multiplying by 0x100, remove the
1827 // <<8 for little-endian, and remove the unshifted value
1828 // for big-endian.
1829 if (color_type & 4) // Alpha channel
1830 *((guint16*)(ptr+2)) = a + (a<<8);
1831 } else if (bit_depth == 8) {
1832 *ptr = guint8(gray >> 8);
1833 if (color_type & 4) // Alpha channel
1834 *((guint8*)(ptr+1)) = a;
1835 } else {
1836 if (!pad) *ptr=0;
1837 // In PNG numbers are stored left to right, but in most significant bits first, so the first one processed is the ``big'' mask, etc.
1838 int realpad = 8 - bit_depth - pad;
1839 *ptr += guint8((gray >> (16-bit_depth))<<realpad); // Note the "+="
1840 if (color_type & 4) // Alpha channel
1841 *(ptr+1) += guint8((a >> (8-bit_depth))<<(bit_depth + realpad));
1842 }
1843 }
1844
1845 pad += bit_depth*n_fields;
1846 ptr += pad/8;
1847 pad %= 8;
1848 }
1849 // Align bytes on rows
1850 if (pad) {
1851 pad = 0;
1852 ptr++;
1853 }
1854 }
1855 return new_data;
1856}
