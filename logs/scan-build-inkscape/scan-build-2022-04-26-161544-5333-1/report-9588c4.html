752static void StyleFromSelectionToTool(Glib::ustring const &prefs_path, StyleSwatch *swatch)
753{
754 SPDesktop *desktop = SP_ACTIVE_DESKTOP((Inkscape::Application::instance()).active_desktop());
755 if (desktop == nullptr)
756 return;
757
758 Inkscape::Selection *selection = desktop->getSelection();
759
760 if (selection->isEmpty()) {
761 desktop->getMessageStack()->flash(Inkscape::ERROR_MESSAGE,
762 _("<b>No objects selected</b> to take the style from.")gettext ("<b>No objects selected</b> to take the style from."
));
763 return;
764 }
765 SPItem *item = selection->singleItem();
766 if (!item) {
767 /* TODO: If each item in the selection has the same style then don't consider it an error.
768 * Maybe we should try to handle multiple selections anyway, e.g. the intersection of the
769 * style attributes for the selected items. */
770 desktop->getMessageStack()->flash(Inkscape::ERROR_MESSAGE,
771 _("<b>More than one object selected.</b> Cannot take style from multiple objects.")gettext ("<b>More than one object selected.</b> Cannot take style from multiple objects."
));
772 return;
773 }
774
775 SPCSSAttr *css = take_style_from_item (item);
776
777 if (!css) return;
778
779 // remove black-listed properties
780 css = sp_css_attr_unset_blacklist (css);
781
782 // only store text style for the text tool
783 if (prefs_path != "/tools/text") {
784 css = sp_css_attr_unset_text (css);
785 }
786
787 // we cannot store properties with uris - they will be invalid in other documents
788 css = sp_css_attr_unset_uris (css);
789
790 Inkscape::Preferences *prefs = Inkscape::Preferences::get();
791 prefs->setStyle(prefs_path + "/style", css);
792 sp_repr_css_attr_unref (css);
793
794 // update the swatch
795 if (swatch) {
796 SPCSSAttr *css = prefs->getInheritedStyle(prefs_path + "/style");
797 swatch->setStyle (css);
798 sp_repr_css_attr_unref(css);
799 }
800}
