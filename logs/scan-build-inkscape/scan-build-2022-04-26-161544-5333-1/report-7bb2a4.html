534Geom::PathVector
535LPEMirrorSymmetry::doEffect_path (Geom::PathVector const & path_in)
536{
537 if (split_items && !fuse_paths) {
538 return path_in;
539 }
540 Geom::PathVector const original_pathv = pathv_to_linear_and_cubic_beziers(path_in);
541 Geom::PathVector path_out;
542
543 if (!discard_orig_path && !fuse_paths) {
544 path_out = pathv_to_linear_and_cubic_beziers(path_in);
545 }
546
547 Geom::Line line_separation((Geom::Point)start_point, (Geom::Point)end_point);
548 Geom::Affine m = Geom::reflection (line_separation.vector(), (Geom::Point)start_point);
549 if (fuse_paths && !discard_orig_path) {
550 for (const auto & path_it : original_pathv)
551 {
552 if (path_it.empty()) {
553 continue;
554 }
555 Geom::PathVector tmp_pathvector;
556 double time_start = 0.0;
557 int position = 0;
558 bool end_open = false;
559 if (path_it.closed()) {
560 const Geom::Curve &closingline = path_it.back_closed();
561 if (!are_near(closingline.initialPoint(), closingline.finalPoint())) {
562 end_open = true;
563 }
564 }
565 Geom::Path original = path_it;
566 if (end_open && path_it.closed()) {
567 original.close(false);
568 original.appendNew<Geom::LineSegment>( original.initialPoint() );
569 original.close(true);
570 }
571 Geom::Point s = start_point;
572 Geom::Point e = end_point;
573 double dir = line_separation.angle();
574 double diagonal = Geom::distance(Geom::Point(boundingbox_X.min(),boundingbox_Y.min()),Geom::Point(boundingbox_X.max(),boundingbox_Y.max()));
575 Geom::Rect bbox(Geom::Point(boundingbox_X.min(),boundingbox_Y.min()),Geom::Point(boundingbox_X.max(),boundingbox_Y.max()));
576 double size_divider = Geom::distance(center_point, bbox) + diagonal;
577 s = Geom::Point::polar(dir,size_divider) + center_point;
578 e = Geom::Point::polar(dir + Geom::rad_from_deg(180),size_divider) + center_point;
579 Geom::Path divider = Geom::Path(s);
580 divider.appendNew<Geom::LineSegment>(e);
581 Geom::Crossings cs = crossings(original, divider);
582 std::vector<double> crossed;
583 for(auto & c : cs) {
584 crossed.push_back(c.ta);
585 }
586 std::sort(crossed.begin(), crossed.end());
587 for (unsigned int i = 0; i < crossed.size(); i++) {
588 double time_end = crossed[i];
589 if (time_start != time_end && time_end - time_start > Geom::EPSILON) {
590 Geom::Path portion = original.portion(time_start, time_end);
591 if (!portion.empty()) {
592 Geom::Point middle = portion.pointAt((double)portion.size()/2.0);
593 position = Geom::sgn(Geom::cross(e - s, middle - s));
594 if (!oposite_fuse) {
595 position *= -1;
596 }
597 if (position == 1) {
598 if (!split_items) {
599 Geom::Path mirror = portion.reversed() * m;
600 mirror.setInitial(portion.finalPoint());
601 portion.append(mirror);
602 if(i != 0) {
603 portion.setFinal(portion.initialPoint());
604 portion.close();
605 }
606 }
607 tmp_pathvector.push_back(portion);
608 }
609 portion.clear();
610 }
611 }
612 time_start = time_end;
613 }
614 position = Geom::sgn(Geom::cross(e - s, original.finalPoint() - s));
615 if (!oposite_fuse) {
616 position *= -1;
617 }
618 if (cs.size()!=0 && (position == 1)) {
619 if (time_start != original.size() && original.size() - time_start > Geom::EPSILON) {
620 Geom::Path portion = original.portion(time_start, original.size());
621 if (!portion.empty()) {
622 portion = portion.reversed();
623 if (!split_items) {
624 Geom::Path mirror = portion.reversed() * m;
625 mirror.setInitial(portion.finalPoint());
626 portion.append(mirror);
627 }
628 portion = portion.reversed();
629 if (!original.closed()) {
630 tmp_pathvector.push_back(portion);
631 } else {
632 if (cs.size() > 1 && tmp_pathvector.size() > 0 && tmp_pathvector[0].size() > 0 ) {
633 if (!split_items) {
634 portion.setFinal(tmp_pathvector[0].initialPoint());
635 portion.setInitial(tmp_pathvector[0].finalPoint());
636 } else {
637 tmp_pathvector[0] = tmp_pathvector[0].reversed();
638 portion = portion.reversed();
639 portion.setInitial(tmp_pathvector[0].finalPoint());
640 }
641 tmp_pathvector[0].append(portion);
642 } else {
643 tmp_pathvector.push_back(portion);
644 }
645 if (lpeversion.param_getSVGValue() < "1.1") {
646 tmp_pathvector[0].close();
647 }
648 }
649 portion.clear();
650 }
651 }
652 }
653 if (!split_open && lpeversion.param_getSVGValue() >= "1.1" && original.closed()) {
654 for (auto &path : tmp_pathvector) {
655 if (!path.closed()) {
656 path.close();
657 }
658 }
659 sp_flatten(tmp_pathvector, fill_oddEven);
660 }
661 if (cs.size() == 0 && position == 1) {
662 tmp_pathvector.push_back(original);
663 if (!split_items) {
664 tmp_pathvector.push_back(original * m);
665 }
666 }
667 path_out.insert(path_out.end(), tmp_pathvector.begin(), tmp_pathvector.end());
668 tmp_pathvector.clear();
669 }
670 } else if (!fuse_paths || discard_orig_path) {
671 for (const auto & i : original_pathv) {
672 path_out.push_back(i * m);
673 }
674 }
675 return path_out;
