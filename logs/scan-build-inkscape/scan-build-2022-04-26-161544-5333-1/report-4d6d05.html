697{
698 auto dlg2 = reinterpret_cast<ExportProgressDialog *>(dlg);
699
700 auto self = dynamic_cast<BatchExport *>(dlg2->get_export_panel());
701
702 if (!self || self->interrupted)
703 return FALSE(0);
704
705 auto current = dlg2->get_current();
706 auto total = dlg2->get_total();
707 if (total > 0) {
708 double completed = current;
709 completed /= static_cast<double>(total);
710
711 value = completed + (value / static_cast<double>(total));
712 }
713
714 auto prg = dlg2->get_progress();
715 prg->set_fraction(value);
716
717 if (self) {
718 self->_prog->set_fraction(value);
719 }
720
721 int evtcount = 0;
722 while ((evtcount < 16) && gdk_events_pending()) {
723 Gtk::Main::iteration(false);
724 evtcount += 1;
725 }
726
727 Gtk::Main::iteration(false);
728 return TRUE(!(0));
