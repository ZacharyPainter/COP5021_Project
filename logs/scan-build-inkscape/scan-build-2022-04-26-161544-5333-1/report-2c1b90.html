134Geom::PathVector
135sp_pathvector_boolop_slice_intersect(Geom::PathVector const &pathva, Geom::PathVector const &pathvb, bool inside, fill_typ fra, fill_typ frb)
136{
137 // This is similar to sp_pathvector_boolop/bool_op_slice, but keeps only edges inside the cutter area.
138 // The code is also based on sp_pathvector_boolop_slice.
139 //
140 // We have two paths on input
141 // - a closed area which is used to cut out pieces from a contour (called area below)
142 // - a contour which is cut into pieces by the border of thr area (called contour below)
143 //
144 // The code below works in the following steps
145 // (a) Convert the area to a shape, so that we can ask the winding number for any point
146 // (b) Add both, the contour and the area to a single shape and intersect them
147 // (c) Find the intersection points between area border and contour (vector toCut)
148 // (d) Split the original contour at the intersection points
149 // (e) check for each contour edge in combined shape if its center is inside the area - if not discard it
150 // (f) create a vector of all inside edges
151 // (g) convert the piece numbers to the piece numbers after applying the cuts
152 // (h) fill a bool vector with information which pieces are in
153 // (i) filter the descr_cmd of the result path with this bool vector
154 //
155 // The main inefficiency here is step (e) because I use a winding function of the area-shape which goes
156 // through the complete edge list for each point I ask for, so effort is n-edges-contour * n-edges-area.
157 // It is tricky to improve this without building into the livarot code.
158 // One way might be to decide at the intersection points which edges touching the intersection points are
159 // in by making a loop through all edges on the intersection vertex. Since this is a directed non intersecting
160 // graph, this should provide sufficient information.
161 // But since I anyway will change this to the new mechanism some time speed is fairly ok, I didn't look into this.
162
163
164 // extract the livarot Paths from the source objects
165 // also get the winding rule specified in the style
166 // Livarot's outline of arcs is broken. So convert the path to linear and cubics only, for which the outline is created correctly.
167 Path *contour_path = Path_for_pathvector(pathv_to_linear_and_cubic_beziers(pathva));
168 Path *area_path = Path_for_pathvector(pathv_to_linear_and_cubic_beziers(pathvb));
169
170 // Shapes from above paths
171 Shape *area_shape = new Shape;
172 Shape *combined_shape = new Shape;
173 Shape *combined_inters = new Shape;
174
175 // Add the area (process to intersection free shape)
176 area_path->ConvertWithBackData(1.0);
177 area_path->Fill(combined_shape, 1);
178
179 // Convert this to a shape with full winding information
180 area_shape->ConvertToShape(combined_shape, frb);
181
182 // Add the contour to the combined path (just add, no winding processing)
183 contour_path->ConvertWithBackData(1.0);
184 contour_path->Fill(combined_shape, 0, true, false, false);
185
186 // Intersect the area and the contour - no fill processing
187 combined_inters->ConvertToShape(combined_shape, fill_justDont);
188
189 // Result path
190 Path *result_path = new Path;
191 result_path->SetBackData(false);
192
193 // Cutting positions for contour
194 std::vector<Path::cut_position> toCut;
195
196 if (combined_inters->hasBackData()) {
197 // should always be the case, but ya never know
198 {
199 for (int i = 0; i < combined_inters->numberOfPoints(); i++) {
200 if (combined_inters->getPoint(i).totalDegree() > 2) {
201 // possibly an intersection
202 // we need to check that at least one edge from the source path is incident to it
203 // before we declare it's an intersection
204 int cb = combined_inters->getPoint(i).incidentEdge[FIRST];
205 int nbOrig = 0;
206 int nbOther = 0;
207 int piece = -1;
208 float t = 0.0;
209 while (cb >= 0 && cb < combined_inters->numberOfEdges()) {
210 if (combined_inters->ebData[cb].pathID == 0) {
211 // the source has an edge incident to the point, get its position on the path
212 piece = combined_inters->ebData[cb].pieceID;
213 if (combined_inters->getEdge(cb).st == i) {
214 t = combined_inters->ebData[cb].tSt;
215 } else {
216 t = combined_inters->ebData[cb].tEn;
217 }
218 nbOrig++;
219 }
220 if (combined_inters->ebData[cb].pathID == 1) {
221 nbOther++; // the cut is incident to this point
222 }
223 cb = combined_inters->NextAt(i, cb);
224 }
225 if (nbOrig > 0 && nbOther > 0) {
226 // point incident to both path and cut: an intersection
227 // note that you only keep one position on the source; you could have degenerate
228 // cases where the source crosses itself at this point, and you wouyld miss an intersection
229 Path::cut_position cutpos;
230 cutpos.piece = piece;
231 cutpos.t = t;
232 toCut.push_back(cutpos);
233 }
234 }
235 }
236 }
237 {
238 // remove the edges from the intersection polygon
239 int i = combined_inters->numberOfEdges() - 1;
240 for (; i >= 0; i--) {
241 if (combined_inters->ebData[i].pathID == 1) {
242 combined_inters->SubEdge(i);
243 } else {
244 const Shape::dg_arete &edge = combined_inters->getEdge(i);
245 const Shape::dg_point &start = combined_inters->getPoint(edge.st);
246 const Shape::dg_point &end = combined_inters->getPoint(edge.en);
247 Geom::Point mid = 0.5 * (start.x + end.x);
248 int wind = area_shape->PtWinding(mid);
249 if (wind == 0) {
250 combined_inters->SubEdge(i);
251 }
252 }
253 }
254 }
255 }
256
257 // create a vector of pieces, which are in the intersection
258 std::vector<Path::cut_position> inside_pieces(combined_inters->numberOfEdges());
259 for (int i = 0; i < combined_inters->numberOfEdges(); i++) {
260 inside_pieces[i].piece = combined_inters->ebData[i].pieceID;
261 // Use the t middle point, this is safe to compare with values from toCut in the presence of roundoff errors
262 inside_pieces[i].t = 0.5 * (combined_inters->ebData[i].tSt + combined_inters->ebData[i].tEn);
263 }
264 std::sort(inside_pieces.begin(), inside_pieces.end(), cmp_cut_position);
265
266 // sort cut positions
267 std::sort(toCut.begin(), toCut.end(), cmp_cut_position);
268
269 // Compute piece ids after ConvertPositionsToMoveTo
270 {
271 int idIncr = 0;
272 std::vector<Path::cut_position>::iterator itPiece = inside_pieces.begin();
273 std::vector<Path::cut_position>::iterator itCut = toCut.begin();
274 while (itPiece != inside_pieces.end()) {
275 while (itCut != toCut.end() && cmp_cut_position(*itCut, *itPiece)) {
276 ++itCut;
277 idIncr += 2;
278 }
279 itPiece->piece += idIncr;
280 ++itPiece;
281 }
282 }
283
284 // Copy the original path to result and cut at the intersection points
285 result_path->Copy(contour_path);
286 result_path->ConvertPositionsToMoveTo(toCut.size(), toCut.data()); // cut where you found intersections
287
288 // Create an array of bools which states which pieces are in
289 std::vector<bool> inside_flags(result_path->descr_cmd.size(), false);
290 for (auto & inside_piece : inside_pieces) {
291 inside_flags[ inside_piece.piece ] = true;
292 // also enable the element -1 to get the MoveTo
293 if (inside_piece.piece >= 1) {
294 inside_flags[ inside_piece.piece - 1 ] = true;
295 }
296 }
297
298#if 0 // CONCEPT TESTING
299 //Check if the inside/outside verdict is consistent - just for testing the concept
300 // Retrieve the pieces
301 int nParts = 0;
302 Path **parts = result_path->SubPaths(nParts, false);
303
304 // Each piece should be either fully in or fully out
305 int iPiece = 0;
306 for (int iPart = 0; iPart < nParts; iPart++) {
307 bool andsum = true;
308 bool orsum = false;
309 for (int iCmd = 0; iCmd < parts[iPart]->descr_cmd.size(); iCmd++, iPiece++) {
310 andsum = andsum && inside_flags[ iPiece ];
311 orsum = andsum || inside_flags[ iPiece ];
312 }
313
314 if (andsum != orsum) {
315 g_warning("Inconsistent inside/outside verdict for part=%d", iPart);
316 }
317 }
318 g_free(parts);
319#endif
320
321 // iterate over the commands of a path and keep those which are inside
322 int iDest = 0;
323 for (int iSrc = 0; iSrc < result_path->descr_cmd.size(); iSrc++) {
324 if (inside_flags[iSrc] == inside) {
325 result_path->descr_cmd[iDest++] = result_path->descr_cmd[iSrc];
326 } else {
327 delete result_path->descr_cmd[iSrc];
328 }
329 }
330 result_path->descr_cmd.resize(iDest);
331
332 delete combined_inters;
333 delete combined_shape;
334 delete area_shape;
335 delete contour_path;
336 delete area_path;
337
338 gchar *result_str = result_path->svg_dump_path();
339 Geom::PathVector outres = Geom::parse_svg_path(result_str);
340 // CONCEPT TESTING g_warning( "%s", result_str );
341 g_free(result_str);
342 delete result_path;
343
344 return outres;
345}

610 curve->transform(clipshape->transform);

620 for (auto iter : item_list) {
621 Geom::PathVector tmp = get_union(root, iter, _from_original_d);
622 if (res.empty()) {

627 }
628 }
629 }

642 if (res.empty()) {

644 } else {
645 res = sp_pathvector_boolop(res, tmp, to_bool_op(bool_op_ex_union), originfill,

647 }
648 }
649 originfill = GetFillTyp(shape);
Value stored to 'originfill' is never read

654 if (curve) {

667 }
668 return res;

671void LPEBool::doEffect(SPCurve *curve)
672{
673 Geom::PathVector path_in = curve->get_pathvector();
674 SPItem *current_operand = dynamic_cast<SPItem *>(operand_item.getObject());
675 if (current_operand == current_shape) {
676 g_warning("operand and current shape are the same");
677 operand_item.param_set_default();
678 return;
679 }
680 if (onremove) {
681 current_operand = dynamic_cast<SPItem *>(getSPDoc()->getObjectById(operand_id));
682 }
683 if (current_operand) {
684 bool_op_ex op = bool_operation.get_value();
685 bool swap = swap_operands.get_value();
686 if (op == bool_op_ex_cut_both) {
687 swap = false;
688 }
689
690 Geom::Affine current_affine = sp_lpe_item->transform;
691 Geom::PathVector operand_pv = get_union(current_operand, current_operand);
692 if (operand_pv.empty()) {
693 return;
694 }
695 path_in *= current_affine;
696
697 Geom::PathVector path_a = swap ? path_in : operand_pv;
698 Geom::PathVector path_b = swap ? operand_pv : path_in;
699 _hp = path_a;
700 _hp.insert(_hp.end(), path_b.begin(), path_b.end());
701 _hp *= current_affine.inverse();
702 auto item = dynamic_cast<SPItem *>(operand_item.getObject());
703 fill_typ fill_this = fill_type_this.get_value() != fill_justDont ? fill_type_this.get_value() : GetFillTyp(current_shape);
704 fill_typ fill_operand =
705 fill_type_operand.get_value() != fill_justDont ? fill_type_operand.get_value() : GetFillTyp(item);
706
707 fill_typ fill_a = swap ? fill_this : fill_operand;
708 fill_typ fill_b = swap ? fill_operand : fill_this;
709
710 if (rmv_inner.get_value()) {
711 path_b = sp_pathvector_boolop_remove_inner(path_b, fill_b);
712 }
713 Geom::PathVector path_out;
714 helperLineSatellites = false;
715 if (op == bool_op_ex_cut) {
716 if (onremove) {
717 path_out = sp_pathvector_boolop(path_a, path_b, to_bool_op(bool_op_ex_diff), fill_a, fill_b, legacytest_livarotonly);
718 } else {
719 int error = 0;
720 Geom::PathVector path_tmp = sp_pathvector_boolop(path_a, path_b, to_bool_op(op), fill_a, fill_b, legacytest_livarotonly, true, error);
721 for (auto pathit : path_tmp) {
722 if (pathit.size() != 2 || !error) {
723 path_out.push_back(pathit);
724 }
725 }
726 }
727 /* } else if (op == bool_op_ex_slice) {
728 path_out = sp_pathvector_boolop_slice_intersect(path_a, path_b, true, fill_a, fill_b);
729 Geom::PathVector path_tmp = sp_pathvector_boolop_slice_intersect(path_a, path_b, false, fill_a, fill_b);
730 for (auto pathit : path_tmp) {
731 path_out.push_back(pathit);
732 }
733 } else if (op == bool_op_ex_slice_inside) {
734 path_out = sp_pathvector_boolop_slice_intersect(path_a, path_b, true, fill_a, fill_b);
735 } else if (op == bool_op_ex_slice_outside) {
736 path_out = sp_pathvector_boolop_slice_intersect(path_a, path_b, false, fill_a, fill_b);
737 */
738 } else if (op == bool_op_ex_cut_both){
739 if (onremove) {
740 path_out = sp_pathvector_boolop(path_a, path_b, to_bool_op(bool_op_ex_diff), fill_a, fill_b, legacytest_livarotonly);
741 } else {
742 helperLineSatellites = true;
743 path_out = sp_pathvector_boolop(path_a, path_b, (bool_op) bool_op_diff, fill_a, fill_b, legacytest_livarotonly);
744 auto tmp = sp_pathvector_boolop(path_a, path_b, (bool_op) bool_op_inters, fill_a, fill_b, legacytest_livarotonly);
745 path_out.insert(path_out.end(),tmp.begin(),tmp.end());
746 /* auto tmp2 = sp_pathvector_boolop(path_a, path_b, (bool_op) bool_op_diff, fill_a, fill_b);
747 path_out.insert(path_out.end(),tmp2.begin(),tmp2.end()); */
748 }
749 } else {
750 path_out = sp_pathvector_boolop(path_a, path_b, (bool_op) op, fill_a, fill_b, legacytest_livarotonly);
751 }
752 curve->set_pathvector(path_out * current_affine.inverse());
753 }

904void LPEBool::doOnRemove(SPLPEItem const * lpeitem)
905{
906 // set "keep paths" hook on sp-lpe-item.cpp
907 remove_filter(operand_item.getObject());
908 SPItem *operand = dynamic_cast<SPItem *>(getSPDoc()->getObjectById(operand_id));
909 if (operand) {
910 if (keep_paths) {
911 bool_op_ex op = bool_operation.get_value();
912 if (op == bool_op_ex_cut || op == bool_op_ex_cut_both) {
913 reverse = lpeitem->pos_in_parent() > operand->pos_in_parent();
914 division = nullptr;
915 Geom::PathVector unionpv = get_union(operand, operand);
916 divisionit(operand, sp_lpe_item, unionpv);
917 onremove = true;
918 sp_lpe_item_update_patheffect(sp_lpe_item, false, true);
919 if (op == bool_op_ex_cut_both) {
920 auto * a = dynamic_cast<SPItem *>(getSPDoc()->getObjectById(division_id));
921 if (a) {
922 unionpv = get_union(sp_lpe_item, sp_lpe_item, true);
923 fractureit(operand, unionpv);
924 auto * b = dynamic_cast<SPItem *>(getSPDoc()->getObjectById(division_other_id));
925 if (a && b) {
926 if (reverse) {
927 b->lowerOne();
928 }
929 }
930 }
931 }
932 // we reset variables because LPE is not removed on undo so I need to get clean on redo
933 division = nullptr;
934 division_both = nullptr;
935 division_other = nullptr;
936 operand_id = "";
937 division_id = "";
938 division_other_id = "";
939 onremove = false;
940 }
941 if (is_visible) {
942 processObjects(LPE_ERASE);
943 }
944 }
945 }
