109void SPLPEItem::set(SPAttr key, gchar const* value) {
110 switch (key) {
111 case SPAttr::INKSCAPE_PATH_EFFECT:
112 {
113 this->current_path_effect = nullptr;
114
115 // Disable the path effects while populating the LPE list
116 sp_lpe_item_enable_path_effects(this, false);
117
118 // disconnect all modified listeners:
119 for (auto & mod_it : *this->lpe_modified_connection_list)
120 {
121 mod_it.disconnect();
122 }
123
124 this->lpe_modified_connection_list->clear();
125 clear_path_effect_list(this->path_effect_list);
126
127 // Parse the contents of "value" to rebuild the path effect reference list
128 if ( value ) {
129 std::istringstream iss(value);
130 std::string href;
131
132 while (std::getline(iss, href, ';'))
133 {
134 std::shared_ptr<Inkscape::LivePathEffect::LPEObjectReference>path_effect_ref = std::make_shared<Inkscape::LivePathEffect::LPEObjectReference>(this);
135
136 try {
137 path_effect_ref->link(href.c_str());
138 } catch (Inkscape::BadURIException &e) {
139 g_warning("BadURIException when trying to find LPE: %s", e.what());
140 path_effect_ref->unlink();
141 //delete path_effect_ref;
142 path_effect_ref = nullptr;
143 }
144
145 this->path_effect_list->push_back(path_effect_ref);
146
147 if ( path_effect_ref->lpeobject && path_effect_ref->lpeobject->get_lpe() ) {
148 // connect modified-listener
149 this->lpe_modified_connection_list->push_back(
150 path_effect_ref->lpeobject->connectModified(sigc::bind(sigc::ptr_fun(&lpeobject_ref_modified), this)) );
151 } else {
152 // on clipboard we fix refs so in middle time of the operation, in LPE with multiples path
153 // effects can result middle updata and fire a warning, so we silent it
154 if (!isOnClipboard()) {
155 // something has gone wrong in finding the right patheffect.
156 g_warning("Unknown LPE type specified, LPE stack effectively disabled");
157 // keep the effect in the lpestack, so the whole stack is effectively disabled but
158 // maintained
159 }
160 }
161 }
162 }
163
164 sp_lpe_item_enable_path_effects(this, true);
165 }
166 break;
167
168 default:
169 SPItem::set(key, value);
170 break;
171 }

653{
654 if (!value.empty()) {
655 // Apply the path effects here because in the casse of a group, lpe->resetDefaults
656 // needs that all the subitems have their effects applied
657 SPGroup *group = dynamic_cast<SPGroup *>(this);
658 if (group) {
659 sp_lpe_item_update_patheffect(this, false, true);
660 }
661 // Disable the path effects while preparing the new lpe
662 sp_lpe_item_enable_path_effects(this, false);
663
664 // Add the new reference to the list of LPE references
665 HRefList hreflist;
666 for (PathEffectList::const_iterator it = this->path_effect_list->begin(); it != this->path_effect_list->end(); ++it)
667 {
668 hreflist.push_back( std::string((*it)->lpeobject_href) );

673 // Make sure that ellipse is stored as <svg:path>
674 if( SP_IS_GENERICELLIPSE(this)) {
675 SP_GENERICELLIPSE(this)->write( this->getRepr()->document(), this->getRepr(), SP_OBJECT_WRITE_EXT(1 << 1) );

680 if (lpeobj && lpeobj->get_lpe()) {
681 Inkscape::LivePathEffect::Effect *lpe = lpeobj->get_lpe();
682 // Ask the path effect to reset itself if it doesn't have parameters yet
683 if (reset) {
684 // has to be called when all the subitems have their lpes applied

688 // if anyone find necessary
689 // make sure there is an original-d for paths!!!
690 sp_lpe_item_create_original_path_recursive(this);

693 }
694
695 //Enable the path effects now that everything is ready to apply the new path effect
696 sp_lpe_item_enable_path_effects(this, true);
697
698 // Apply the path effect
699 sp_lpe_item_update_patheffect(this, true, true);

1442void SPLPEItem::replacePathEffects( std::vector<LivePathEffectObject const *> const &old_lpeobjs,
1443 std::vector<LivePathEffectObject const *> const &new_lpeobjs )
1444{

1450

1452 std::vector<LivePathEffectObject const *>::difference_type found_index = std::distance (old_lpeobjs.begin(), found_it);

1454 gchar *hrefstr = g_strdup_printf("#%s", repr_id);
1455 hreflist.push_back( std::string(hrefstr) );
1456 g_free(hrefstr);

1466/**
1467 * Check all effects in the stack if they are used by other items, and fork them if so.
1468 * It is not recommended to fork the effects by yourself calling LivePathEffectObject::fork_private_if_necessary,
1469 * use this method instead.
1470 * Returns true if one or more effects were forked; returns false if nothing was done.
1471 */
1472bool SPLPEItem::forkPathEffectsIfNecessary(unsigned int nr_of_allowed_users, bool recursive)
1473{
1474 bool forked = false;
1475 SPGroup * group = dynamic_cast<SPGroup *>(this);
1476 if (group && recursive) {
1477 std::vector<SPItem*> item_list = sp_item_group_item_list(group);
1478 for (auto child:item_list) {
1479 SPLPEItem *lpeitem = dynamic_cast<SPLPEItem *>(child);
1480 if (lpeitem && lpeitem->forkPathEffectsIfNecessary(nr_of_allowed_users, recursive)) {
1481 forked = true;
1482 }
1483 }
1484 }
1485
1486 if ( this->hasPathEffect() ) {
1487 // If one of the path effects is used by 2 or more items, fork it
1488 // so that each object has its own independent copy of the effect.
1489 // Note: replacing path effects messes up the path effect list
1490
1491 // Clones of the LPEItem will increase the refcount of the lpeobjects.
1492 // Therefore, nr_of_allowed_users should be increased with the number of clones (i.e. refs to the lpeitem)
1493 nr_of_allowed_users += this->hrefcount;
1494
1495 std::vector<LivePathEffectObject const *> old_lpeobjs, new_lpeobjs;
1496 PathEffectList effect_list = this->getEffectList();
1497 for (auto & it : effect_list)
1498 {
1499 LivePathEffectObject *lpeobj = it->lpeobject;
1500 if (lpeobj) {
1501 LivePathEffectObject *forked_lpeobj = lpeobj->fork_private_if_necessary(nr_of_allowed_users);
1502 if (forked_lpeobj && forked_lpeobj != lpeobj) {
1503 forked = true;
1504 forked_lpeobj->get_lpe()->is_load = true;
1505 old_lpeobjs.push_back(lpeobj);
1506 new_lpeobjs.push_back(forked_lpeobj);
1507 }
1508 }
1509 }
1510
1511 if (forked) {
1512 this->replacePathEffects(old_lpeobjs, new_lpeobjs);
1513 }
1514 }
1515
1516 return forked;
1517}
