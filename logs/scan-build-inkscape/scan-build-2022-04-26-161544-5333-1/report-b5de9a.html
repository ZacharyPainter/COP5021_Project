216void dump_ustr(Glib::ustring const &ustr)
217{
218 char const *cstr = ustr.c_str();
219 char const *data = ustr.data();
220 Glib::ustring::size_type const byteLen = ustr.bytes();
221 Glib::ustring::size_type const dataLen = ustr.length();
222 Glib::ustring::size_type const cstrLen = strlen(cstr);
223
224 g_message(" size: %lu\n length: %lu\n bytes: %lu\n clen: %lu",
225 gulong(ustr.size()), gulong(dataLen), gulong(byteLen), gulong(cstrLen) );
226 g_message( " ASCII? %s", (ustr.is_ascii() ? "yes":"no") );
227 g_message( " UTF-8? %s", (ustr.validate() ? "yes":"no") );
228
229 try {
230 Glib::ustring tmp;
231 for (Glib::ustring::size_type i = 0; i < ustr.bytes(); i++) {
232 tmp = " ";
233 if (i < dataLen) {
234 Glib::ustring::value_type val = ustr.at(i);
235 gchar* tmp2 = g_strdup_printf( (((val & 0xff00) == 0) ? " %02x" : "%04x"), val );
236 tmp += tmp2;
237 g_free( tmp2 );
238 } else {
239 tmp += " ";
240 }
241
242 if (i < byteLen) {
243 int val = (0x0ff & data[i]);
244 gchar *tmp2 = g_strdup_printf(" %02x", val);
245 tmp += tmp2;
246 g_free( tmp2 );
247 if ( val > 32 && val < 127 ) {
248 tmp2 = g_strdup_printf( " '%c'", (gchar)val );
249 tmp += tmp2;
250 g_free( tmp2 );
251 } else {
252 tmp += " . ";
253 }
254 } else {
255 tmp += " ";
256 }
257
258 if ( i < cstrLen ) {
259 int val = (0x0ff & cstr[i]);
260 gchar* tmp2 = g_strdup_printf(" %02x", val);
261 tmp += tmp2;
262 g_free(tmp2);
263 if ( val > 32 && val < 127 ) {
264 tmp2 = g_strdup_printf(" '%c'", (gchar) val);
265 tmp += tmp2;
266 g_free( tmp2 );
267 } else {
268 tmp += " . ";
269 }
270 } else {
271 tmp += " ";
272 }
273
274 g_message( "%s", tmp.c_str() );
275 }
276 } catch (...) {
277 g_message("XXXXXXXXXXXXXXXXXX Exception" );
278 }
279 g_message("---------------");
280}

"'%s'")
521 "The following additional information was returned by the output extension:\n"gettext ("File %s could not be saved.\n\n" "The following additional information was returned by the output extension:\n"
"'%s'")
522 "'%s'")gettext ("File %s could not be saved.\n\n" "The following additional information was returned by the output extension:\n"
"'%s'"), safeUri, e.what());
523 SP_ACTIVE_DESKTOP((Inkscape::Application::instance()).active_desktop())->messageStack()->flash(Inkscape::ERROR_MESSAGE, _("Document not saved.")gettext ("Document not saved."));

528 return false;
529 } catch (...) {
530 g_critical("Extension '%s' threw an unspecified exception.", key->get_id());
531 gchar *safeUri = Inkscape::IO::sanitizeString(uri.c_str());

535 g_free(text);
536 g_free(safeUri);
537 doc->getReprRoot()->setAttribute("inkscape:version", sp_version_to_string( save ));
538 return false;
539 }
540
541 if (SP_ACTIVE_DESKTOP((Inkscape::Application::instance()).active_desktop())) {
542 if (! SP_ACTIVE_DESKTOP((Inkscape::Application::instance()).active_desktop())->messageStack()) {

556 SP_ACTIVE_DESKTOP((Inkscape::Application::instance()).active_desktop())->messageStack()->flash(Inkscape::NORMAL_MESSAGE, msg.c_str());
557 return true;

561/**

563 */
564bool
565sp_file_save_dialog(Gtk::Window &parentWindow, SPDocument *doc, Inkscape::Extension::FileSaveMethod save_method)
566{
567 Inkscape::Extension::Output *extension = nullptr;
568 bool is_copy = (save_method == Inkscape::Extension::FILE_SAVE_METHOD_SAVE_COPY);
569
570 // Note: default_extension has the format "org.inkscape.output.svg.inkscape", whereas
571 // filename_extension only uses ".svg"
572 Glib::ustring default_extension;
573 Glib::ustring filename_extension = ".svg";
574
575 default_extension= Inkscape::Extension::get_file_save_extension(save_method);
576 //g_message("%s: extension name: '%s'", __FUNCTION__, default_extension);
577
578 extension = dynamic_cast<Inkscape::Extension::Output *>
579 (Inkscape::Extension::db.get(default_extension.c_str()));
580
581 if (extension)
582 filename_extension = extension->get_extension();
583
584 Glib::ustring save_path = Inkscape::Extension::get_file_save_path(doc, save_method);
585
586 if (!Inkscape::IO::file_test(save_path.c_str(),
587 (GFileTest)(G_FILE_TEST_EXISTS | G_FILE_TEST_IS_DIR)))
588 save_path.clear();
589
590 if (save_path.empty())
591 save_path = g_get_home_dir();
592
593 Glib::ustring save_loc = save_path;
594 save_loc.append(G_DIR_SEPARATOR_S"/");
595
596 int i = 1;
597 if ( !doc->getDocumentFilename() ) {
598 // We are saving for the first time; create a unique default filename
599 save_loc = save_loc + _("drawing")gettext ("drawing") + filename_extension;
600
601 while (Inkscape::IO::file_test(save_loc.c_str(), G_FILE_TEST_EXISTS)) {
602 save_loc = save_path;
603 save_loc.append(G_DIR_SEPARATOR_S"/");
604 save_loc = save_loc + Glib::ustring::compose(_("drawing-%1")gettext ("drawing-%1"), i++) + filename_extension;
605 }
606 } else {
607 save_loc.append(Glib::path_get_basename(doc->getDocumentFilename()));
608 }
609
610 // convert save_loc from utf-8 to locale
611 // is this needed any more, now that everything is handled in
612 // Inkscape::IO?
613 Glib::ustring save_loc_local = Glib::filename_from_utf8(save_loc);
614
615 if (!save_loc_local.empty())
616 save_loc = save_loc_local;
617
618 //# Show the SaveAs dialog
619 char const * dialog_title;
620 if (is_copy) {
621 dialog_title = (char const *) _("Select file to save a copy to")gettext ("Select file to save a copy to");
622 } else {
623 dialog_title = (char const *) _("Select file to save to")gettext ("Select file to save to");
624 }
625 gchar* doc_title = doc->getRoot()->title();
626 Inkscape::UI::Dialog::FileSaveDialog *saveDialog =
627 Inkscape::UI::Dialog::FileSaveDialog::create(
628 parentWindow,
629 save_loc,
630 Inkscape::UI::Dialog::SVG_TYPES,
631 dialog_title,
632 default_extension,
633 doc_title ? doc_title : "",
634 save_method
635 );
636
637 saveDialog->setSelectionType(extension);
638
639 bool success = saveDialog->show();
640 if (!success) {
641 delete saveDialog;
642 if(doc_title) g_free(doc_title);
643 return success;
644 }
645
646 // set new title here (call RDF to ensure metadata and title element are updated)
647 rdf_set_work_entity(doc, rdf_find_entity("title"), saveDialog->getDocTitle().c_str());
648
649 Glib::ustring fileName = saveDialog->getFilename();
650 Inkscape::Extension::Extension *selectionType = saveDialog->getSelectionType();
651
652 delete saveDialog;
653 saveDialog = nullptr;
654 if(doc_title) g_free(doc_title);
655
656 if (!fileName.empty()) {
657 Glib::ustring newFileName = Glib::filename_to_utf8(fileName);
658
659 if (!newFileName.empty())
660 fileName = newFileName;
661 else
662 g_warning( "Error converting filename for saving to UTF-8." );
663
664 // FIXME: does the argument !is_copy really convey the correct meaning here?
665 success = file_save(parentWindow, doc, fileName, selectionType, TRUE(!(0)), !is_copy, save_method);
666
667 if (success && doc->getDocumentFilename()) {
668 // getDocumentFilename does not return an actual filename... it is an UTF-8 encoded filename (!)
669 std::string filename = Glib::filename_from_utf8(doc->getDocumentFilename());
670 Glib::ustring uri = Glib::filename_to_uri(filename);
671
672 Glib::RefPtr<Gtk::RecentManager> recent = Gtk::RecentManager::get_default();
673 recent->add_item(uri);
674 }
675
676 save_path = Glib::path_get_dirname(fileName);
677 Inkscape::Extension::store_save_path_in_prefs(save_path, save_method);
678
679 return success;
680 }
681
682
683 return false;
684}

687/**
688 * Save a document, displaying a SaveAs dialog if necessary.
689 */
690bool
691sp_file_save_document(Gtk::Window &parentWindow, SPDocument *doc)
692{
693 bool success = true;
694
695 if (doc->isModifiedSinceSave()) {
696 if ( doc->getDocumentFilename() == nullptr )
697 {
698 // In this case, an argument should be given that indicates that the document is the first
699 // time saved, so that .svg is selected as the default and not the last one "Save as ..." extension used
700 return sp_file_save_dialog(parentWindow, doc, Inkscape::Extension::FILE_SAVE_METHOD_INKSCAPE_SVG);
701 } else {
702 Glib::ustring extension = Inkscape::Extension::get_file_save_extension(Inkscape::Extension::FILE_SAVE_METHOD_SAVE_AS);
703 Glib::ustring fn = g_strdup(doc->getDocumentFilename());
704 // Try to determine the extension from the filename; this may not lead to a valid extension,
705 // but this case is caught in the file_save method below (or rather in Extension::save()
706 // further down the line).

708 Glib::ustring::size_type pos = fn.rfind('.');
709 if (pos != Glib::ustring::npos) {
710 // FIXME: this could/should be more sophisticated (see FileSaveDialog::appendExtension()),
711 // but hopefully it's a reasonable workaround for now
712 ext = fn.substr( pos );
713 }
714 success = file_save(parentWindow, doc, fn, Inkscape::Extension::db.get(ext.c_str()), FALSE(0), TRUE(!(0)), Inkscape::Extension::FILE_SAVE_METHOD_SAVE_AS);

716 // give the user the chance to change filename or extension
717 return sp_file_save_dialog(parentWindow, doc, Inkscape::Extension::FILE_SAVE_METHOD_INKSCAPE_SVG);
718 }
719 }
720 } else {
721 Glib::ustring msg;
722 if (doc->getDocumentFilename() == nullptr )
723 {
724 msg = Glib::ustring::format(_("No changes need to be saved.")gettext ("No changes need to be saved."));
725 } else {
726 msg = Glib::ustring::format(_("No changes need to be saved.")gettext ("No changes need to be saved."), " ", doc->getDocumentFilename());
727 }
728 SP_ACTIVE_DESKTOP((Inkscape::Application::instance()).active_desktop())->messageStack()->flash(Inkscape::WARNING_MESSAGE, msg.c_str());
729 success = TRUE(!(0));
730 }
731
732 return success;
733}

888/**
889 * Paste the contents of a document into the active desktop.
890 * @param clipdoc The document to paste
891 * @param in_place Whether to paste the selection where it was when copied
892 * @pre @c clipdoc is not empty and items can be added to the current layer
893 */
894void sp_import_document(SPDesktop *desktop, SPDocument *clipdoc, bool in_place)
895{
896 //TODO: merge with file_import()
897
898 SPDocument *target_document = desktop->getDocument();
899 Inkscape::XML::Node *root = clipdoc->getReprRoot();
900 Inkscape::XML::Node *target_parent = desktop->layerManager().currentLayer()->getRepr();
901
902 Inkscape::Preferences *prefs = Inkscape::Preferences::get();
903
904 auto *node_after = desktop->getSelection()->topRepr();
905 if (node_after && prefs->getBool("/options/paste/aboveselected", true)) {
906 target_parent = node_after->parent();
907 } else {
908 node_after = target_parent->lastChild();
909 }
910
911 // copy definitions
912 desktop->doc()->importDefs(clipdoc);
913
914 Inkscape::XML::Node* clipboard = nullptr;
915 // copy objects
916 std::vector<Inkscape::XML::Node*> pasted_objects;
917 for (Inkscape::XML::Node *obj = root->firstChild() ; obj ; obj = obj->next()) {
918 // Don't copy metadata, defs, named views and internal clipboard contents to the document
919 if (!strcmp(obj->name(), "svg:defs")) {
920 continue;
921 }
922 if (!strcmp(obj->name(), "svg:metadata")) {
923 continue;
924 }
925 if (!strcmp(obj->name(), "sodipodi:namedview")) {
926 continue;
927 }
928 if (!strcmp(obj->name(), "inkscape:clipboard")) {
929 clipboard = obj;
930 continue;
931 }
932
933 Inkscape::XML::Node *obj_copy = obj->duplicate(target_document->getReprDoc());
934 target_parent->addChild(obj_copy, node_after);
935 node_after = obj_copy;
936 Inkscape::GC::release(obj_copy);
937
938 pasted_objects.push_back(obj_copy);
939
940 // if we are pasting a clone to an already existing object, its
941 // transform is relative to the document, not to its original (see ui/clipboard.cpp)
942 SPUse *use = dynamic_cast<SPUse *>(obj_copy);
943 if (use) {
944 SPItem *original = use->get_original();
945 if (original) {
946 Geom::Affine relative_use_transform = original->transform.inverse() * use->transform;
947 obj_copy->setAttributeOrRemoveIfEmpty("transform", sp_svg_transform_write(relative_use_transform));
948 }
949 }
950 }
951
952 std::vector<Inkscape::XML::Node*> pasted_objects_not;
953 auto layer = desktop->layerManager().currentLayer();
954 Geom::Affine doc2parent = layer->i2doc_affine().inverse();
955
956 if (clipboard) {
957 for (Inkscape::XML::Node *obj = clipboard->firstChild(); obj; obj = obj->next()) {
958 if (target_document->getObjectById(obj->attribute("id")))
959 continue;
960 Inkscape::XML::Node *obj_copy = obj->duplicate(target_document->getReprDoc());
961 layer->appendChildRepr(obj_copy);
962 Inkscape::GC::release(obj_copy);
963 pasted_objects_not.push_back(obj_copy);
964 }
965 }
966 target_document->ensureUpToDate();
967 Inkscape::Selection *selection = desktop->getSelection();
968 selection->setReprList(pasted_objects_not);
969
970 selection->deleteItems();
971
972 // Change the selection to the freshly pasted objects
973 selection->setReprList(pasted_objects);
974 for (auto item : selection->items()) {
975 SPLPEItem *pasted_lpe_item = dynamic_cast<SPLPEItem *>(item);
976 if (pasted_lpe_item) {
977 sp_lpe_item_enable_path_effects(pasted_lpe_item, false);
978 }
979 }
980 // Apply inverse of parent transform
981 selection->applyAffine(desktop->dt2doc() * doc2parent * desktop->doc2dt(), true, false, false);
982
983
984
985 // Update (among other things) all curves in paths, for bounds() to work
986 target_document->ensureUpToDate();
987
988 // move selection either to original position (in_place) or to mouse pointer
989 Geom::OptRect sel_bbox = selection->visualBounds();
990 if (sel_bbox) {
991 // get offset of selection to original position of copied elements
992 Geom::Point pos_original;
993 Inkscape::XML::Node *clipnode = sp_repr_lookup_name(root, "inkscape:clipboard", 1);
994 if (clipnode) {
995 Geom::Point min, max;
996 min = clipnode->getAttributePoint("min", min);
997 max = clipnode->getAttributePoint("max", max);
998 pos_original = Geom::Point(min[Geom::X], max[Geom::Y]);
999 }
1000 Geom::Point offset = pos_original - sel_bbox->corner(3);
1001
1002 if (!in_place) {
1003 SnapManager &m = desktop->namedview->snap_manager;
1004 m.setup(desktop);
1005 desktop->event_context->discard_delayed_snap_event();
1006
1007 // get offset from mouse pointer to bbox center, snap to grid if enabled
1008 Geom::Point mouse_offset = desktop->point() - sel_bbox->midpoint();
1009 offset = m.multipleOfGridPitch(mouse_offset - offset, sel_bbox->midpoint() + offset) + offset;
1010 m.unSetup();
1011 }
1012
1013 selection->moveRelative(offset);
1014 for (auto po : pasted_objects) {
1015 SPLPEItem *lpeitem = dynamic_cast<SPLPEItem *>(target_document->getObjectByRepr(po));
1016 if (lpeitem) {
1017 sp_lpe_item_enable_path_effects(lpeitem, true);
1018 }
1019 }
1020 }
1021 target_document->emitReconstructionFinish();
1022}

1025/**
1026 * Import a resource. Called by sp_file_import()
1027 */
1028SPObject *
1029file_import(SPDocument *in_doc, const Glib::ustring &uri,
1030 Inkscape::Extension::Extension *key)
1031{
1032 SPDesktop *desktop = SP_ACTIVE_DESKTOP((Inkscape::Application::instance()).active_desktop());
1033 bool cancelled = false;
1034 auto prefs = Inkscape::Preferences::get();
1035 bool onimport = prefs->getBool("/options/onimport", true);
1036
1037 // store mouse pointer location before opening any dialogs, so we can drop the item where initially intended
1038 auto pointer_location = desktop->point();
1039
1040 //DEBUG_MESSAGE( fileImport, "file_import( in_doc:%p uri:[%s], key:%p", in_doc, uri, key );
1041 SPDocument *doc;
1042 try {
1043 doc = Inkscape::Extension::open(key, uri.c_str());
1044 } catch (Inkscape::Extension::Input::no_extension_found &e) {
1045 doc = nullptr;
1046 } catch (Inkscape::Extension::Input::open_failed &e) {
1047 doc = nullptr;
1048 } catch (Inkscape::Extension::Input::open_cancelled &e) {
1049 doc = nullptr;
1050 cancelled = true;
1051 }
1052
1053 if (onimport && !prefs->getBool("/options/onimport", true)) {
1054 // Opened instead of imported (onimport set to false in Svg::open)
1055 prefs->setBool("/options/onimport", true);
1056 return nullptr;
1057 } else if (doc != nullptr) {
1058 // Always preserve any imported text kerning / formatting
1059 auto root_repr = in_doc->getReprRoot();
1060 root_repr->setAttribute("xml:space", "preserve");
1061
1062 Inkscape::XML::rebase_hrefs(doc, in_doc->getDocumentBase(), false);
1063 Inkscape::XML::Document *xml_in_doc = in_doc->getReprDoc();
1064 prevent_id_clashes(doc, in_doc, true);
1065 sp_file_fix_lpe(doc);
1066
1067 in_doc->importDefs(doc);
1068
1069 // The extension should set it's pages enabled or disabled when opening
1070 // in order to indicate if pages are being imported or if objects are.
1071 if (doc->getPageManager().hasPages()) {
1072 file_import_pages(in_doc, doc);
1073 DocumentUndo::done(in_doc, _("Import Pages")gettext ("Import Pages"), INKSCAPE_ICON("document-import")"document-import");
1074 // This return is only used by dbus in document-interface.cpp (now removed).
1075 return nullptr;
1076 }
1077
1078 SPCSSAttr *style = sp_css_attr_from_object(doc->getRoot());
1079
1080 // Count the number of top-level items in the imported document.
1081 guint items_count = 0;
1082 SPObject *o = nullptr;
1083 for (auto& child: doc->getRoot()->children) {
1084 if (SP_IS_ITEM(&child)) {
1085 items_count++;
1086 o = &child;
1087 }
1088 }
1089
1090 //ungroup if necessary
1091 bool did_ungroup = false;
1092 while(items_count==1 && o && SP_IS_GROUP(o) && o->children.size()==1){
1093 std::vector<SPItem *>v;
1094 sp_item_group_ungroup(SP_GROUP(o),v,false);
1095 o = v.empty() ? nullptr : v[0];
1096 did_ungroup=true;
1097 }
1098
1099 // Create a new group if necessary.
1100 Inkscape::XML::Node *newgroup = nullptr;
1101 const auto & al = style->attributeList();
1102 if ((style && !al.empty()) || items_count > 1) {
1103 newgroup = xml_in_doc->createElement("svg:g");
1104 sp_repr_css_set(newgroup, style, "style");
1105 }
1106
1107 // Determine the place to insert the new object.
1108 // This will be the current layer, if possible.
1109 // FIXME: If there's no desktop (command line run?) we need
1110 // a document:: method to return the current layer.
1111 // For now, we just use the root in this case.
1112 SPObject *place_to_insert;
1113 if (desktop) {
1114 place_to_insert = desktop->layerManager().currentLayer();
1115 } else {
1116 place_to_insert = in_doc->getRoot();
1117 }
1118
1119 // Construct a new object representing the imported image,
1120 // and insert it into the current document.
1121 SPObject *new_obj = nullptr;
1122 for (auto& child: doc->getRoot()->children) {
1123 if (SP_IS_ITEM(&child)) {
1124 Inkscape::XML::Node *newitem = did_ungroup ? o->getRepr()->duplicate(xml_in_doc) : child.getRepr()->duplicate(xml_in_doc);
1125
1126 // convert layers to groups, and make sure they are unlocked
1127 // FIXME: add "preserve layers" mode where each layer from
1128 // import is copied to the same-named layer in host
1129 newitem->removeAttribute("inkscape:groupmode");
1130 newitem->removeAttribute("sodipodi:insensitive");
1131
1132 if (newgroup) newgroup->appendChild(newitem);
1133 else new_obj = place_to_insert->appendChildRepr(newitem);
1134 }
1135
1136 // don't lose top-level defs or style elements
1137 else if (child.getRepr()->type() == Inkscape::XML::NodeType::ELEMENT_NODE) {
1138 const gchar *tag = child.getRepr()->name();
1139 if (!strcmp(tag, "svg:style")) {
1140 in_doc->getRoot()->appendChildRepr(child.getRepr()->duplicate(xml_in_doc));
1141 }
1142 }
1143 }
1144 in_doc->emitReconstructionFinish();
1145 if (newgroup) new_obj = place_to_insert->appendChildRepr(newgroup);
1146
1147 // release some stuff
1148 if (newgroup) Inkscape::GC::release(newgroup);
1149 if (style) sp_repr_css_attr_unref(style);
1150
1151 // select and move the imported item
1152 if (new_obj && SP_IS_ITEM(new_obj)) {
1153 Inkscape::Selection *selection = desktop->getSelection();
1154 selection->set(SP_ITEM(new_obj));
1155
1156 // preserve parent and viewBox transformations
1157 // c2p is identity matrix at this point unless ensureUpToDate is called
1158 doc->ensureUpToDate();
1159 Geom::Affine affine = doc->getRoot()->c2p * SP_ITEM(place_to_insert)->i2doc_affine().inverse();
1160 selection->applyAffine(desktop->dt2doc() * affine * desktop->doc2dt(), true, false, false);
1161
1162 // move to mouse pointer
1163 {
1164 desktop->getDocument()->ensureUpToDate();
1165 Geom::OptRect sel_bbox = selection->visualBounds();
1166 if (sel_bbox) {
1167 Geom::Point m( pointer_location - sel_bbox->midpoint() );
1168 selection->moveRelative(m, false);
1169 }
1170 }
1171 }
1172
1173 DocumentUndo::done(in_doc, _("Import")gettext ("Import"), INKSCAPE_ICON("document-import")"document-import");
1174 return new_obj;
1175 } else if (!cancelled) {
1176 gchar *text = g_strdup_printf(_("Failed to load the requested file %s")gettext ("Failed to load the requested file %s"), uri.c_str());
1177 sp_ui_error_dialog(text);
1178 g_free(text);
1179 }
1180
1181 return nullptr;
1182}

1203/**
1204 * Display an Open dialog, import a resource if OK pressed.
1205 */
1206void
1207sp_file_import(Gtk::Window &parentWindow)
1208{
1209 static Glib::ustring import_path;
1210
1211 SPDocument *doc = SP_ACTIVE_DOCUMENT((Inkscape::Application::instance()).active_document());
1212 if (!doc)
1213 return;
1214
1215 Inkscape::Preferences *prefs = Inkscape::Preferences::get();
1216
1217 if(import_path.empty())
1218 {
1219 Glib::ustring attr = prefs->getString("/dialogs/import/path");
1220 if (!attr.empty()) import_path = attr;
1221 }
1222
1223 //# Test if the import_path directory exists
1224 if (!Inkscape::IO::file_test(import_path.c_str(),
1225 (GFileTest)(G_FILE_TEST_EXISTS | G_FILE_TEST_IS_DIR)))
1226 import_path = "";
1227
1228 //# If no open path, default to our home directory
1229 if (import_path.empty())
1230 {
1231 import_path = g_get_home_dir();
1232 import_path.append(G_DIR_SEPARATOR_S"/");
1233 }
1234
1235 // Create new dialog (don't use an old one, because parentWindow has probably changed)
1236 Inkscape::UI::Dialog::FileOpenDialog *importDialogInstance =
1237 Inkscape::UI::Dialog::FileOpenDialog::create(
1238 parentWindow,
1239 import_path,
1240 Inkscape::UI::Dialog::IMPORT_TYPES,
1241 (char const *)_("Select file to import")gettext ("Select file to import"));
1242
1243 bool success = importDialogInstance->show();
1244 if (!success) {
1245 delete importDialogInstance;
1246 return;
1247 }
1248
1249 typedef std::vector<Glib::ustring> pathnames;
1250 pathnames flist(importDialogInstance->getFilenames());
1251
1252 // Get file name and extension type
1253 Glib::ustring fileName = importDialogInstance->getFilename();
1254 Inkscape::Extension::Extension *selection = importDialogInstance->getSelectionType();
1255
1256 delete importDialogInstance;
1257 importDialogInstance = nullptr;
1258
1259 //# Iterate through filenames if more than 1
1260 if (flist.size() > 1)
1261 {
1262 for (const auto & i : flist)
1263 {
1264 fileName = i;
1265
1266 Glib::ustring newFileName = Glib::filename_to_utf8(fileName);
1267 if (!newFileName.empty())
1268 fileName = newFileName;
1269 else
1270 g_warning("ERROR CONVERTING IMPORT FILENAME TO UTF-8");
1271
1272#ifdef INK_DUMP_FILENAME_CONV
1273 g_message("Importing File %s\n", fileName.c_str());
1274#endif
1275 file_import(doc, fileName, selection);
1276 }
1277
1278 return;
1279 }
1280
1281
1282 if (!fileName.empty()) {
1283
1284 Glib::ustring newFileName = Glib::filename_to_utf8(fileName);
1285
1286 if (!newFileName.empty())
1287 fileName = newFileName;
1288 else
1289 g_warning("ERROR CONVERTING IMPORT FILENAME TO UTF-8");
1290
1291 import_path = Glib::path_get_dirname(fileName);
1292 import_path.append(G_DIR_SEPARATOR_S"/");
1293 prefs->setString("/dialogs/import/path", import_path);
1294
1295 file_import(doc, fileName, selection);
1296 }
1297
1298 return;
1299}
