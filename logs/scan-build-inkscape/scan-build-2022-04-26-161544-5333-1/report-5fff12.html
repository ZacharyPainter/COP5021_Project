204/**
205 * Update window width and height in order to fit all dialogs inisde its container.
206 *
207 * The intended use of this function is at initialization.
208 */
209void DialogWindow::update_window_size_to_fit_children()
210{
211 // Declare variables
212 int pos_x = 0, pos_y = 0;
213 int width = 0, height = 0;
214 int overhead = 0, baseline;
215 Gtk::Allocation allocation;
216 Gtk::Requisition minimum_size, natural_size;
217
218 // Read needed data
219 get_position(pos_x, pos_y);
220 get_allocated_size(allocation, baseline);
221 const std::multimap<Glib::ustring, DialogBase *> *dialogs = _container->get_dialogs();
222
223 // Get largest sizes for dialogs
224 for (auto dialog : *dialogs) {
225 dialog.second->get_preferred_size(minimum_size, natural_size);
226 width = std::max(natural_size.width, width);
227 height = std::max(natural_size.height, height);
228 overhead = std::max(overhead, dialog.second->property_margin().get_value());
229 }
230
231 // Compute sizes including overhead
232 overhead = 2 * (WINDOW_DROPZONE_SIZE_LARGE + overhead);
233 width = width + overhead;
234 height = height + overhead + NOTEBOOK_TAB_HEIGHT;
235
236 // If sizes are lower then current, don't change them
237 if (allocation.get_width() >= width && allocation.get_height() >= height) {
238 return;
239 }
240
241 // Compute largest sizes on both axis
242 width = std::max(width, allocation.get_width());
243 height = std::max(height, allocation.get_height());
244
245 // Compute new positions to keep window centered
246 pos_x = pos_x - (width - allocation.get_width()) / 2;
247 pos_y = pos_y - (height - allocation.get_height()) / 2;
248
249 // Keep window inside the screen
250 pos_x = std::max(pos_x, 0);
251 pos_y = std::max(pos_y, 0);
252
253 // Resize window
254 move(pos_x, pos_y);
255 resize(width, height);
256}

258// mimic InkscapeWindow handling of shortcuts to make them work with active floating dialog window
259bool DialogWindow::on_key_press_event(GdkEventKey *key_event)
260{
261 auto focus = get_focus();
262 if (focus) {
263 if (focus->event(reinterpret_cast<GdkEvent *>(key_event))) {
264 return true;
265 }
266 }
267
268 // Pass key event to this window or to app (via this window).
269 if (Gtk::Window::on_key_press_event(key_event)) {
270 return true;
271 }
272
273 // Pass key event to active InkscapeWindow to handle win (and app) level shortcuts.
274 if (_app->get_active_window()->on_key_press_event(key_event)) {
275 return true;
276 }
277
278 return false;
279}
