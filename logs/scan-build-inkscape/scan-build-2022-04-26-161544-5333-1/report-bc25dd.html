1438 case GDK_KEY_KP_Delete0xff9f:

| GDK_MOD1_MASK)) == GDK_CONTROL_MASK));

1443 default:

1445 }
1446 break;
1447 }

1451 case GDK_KEY_Shift_L0xffe1:

1459 break;
1460 default:
1461 sp_tweak_switch_mode (this, prefs->getInt("/tools/tweak/mode"), MOD__SHIFT(event)((event)->key.state & GDK_SHIFT_MASK));
1462 break;
1463 }
1464 }
1465 default:

1467 }
