387Glib::ustring Find::find_replace(const gchar *str, const gchar *find, const gchar *replace, bool exact, bool casematch, bool replaceall)
388{
389 Glib::ustring ustr = str;
390 Glib::ustring ufind = find;
391 if (!casematch) {
392 ufind = ufind.lowercase();
393 }
394 gsize n = find_strcmp_pos(ustr.c_str(), ufind.c_str(), exact, casematch);
395 while (n != std::string::npos) {
396 ustr.replace(n, ufind.length(), replace);
397 if (!replaceall) {
398 return ustr;
399 }
400 // Start the next search after the last replace character to avoid infinite loops (replace "a" with "aaa" etc)
401 n = find_strcmp_pos(ustr.c_str(), ufind.c_str(), exact, casematch, n + strlen(replace) + 1);
402 }
403 return ustr;
404}

406gsize Find::find_strcmp_pos(const gchar *str, const gchar *find, bool exact, bool casematch, gsize start/*=0*/)
407{

409 Glib::ustring ufind = find;
410
411 if (!casematch) {
412 ustr = ustr.lowercase();
413 ufind = ufind.lowercase();
414 }
415
416 gsize pos = std::string::npos;
417 if (exact) {
418 if (ustr == ufind) {
419 pos = 0;
420 }
421 } else {
422 pos = ustr.find(ufind, start);
423 }
424
425 return pos;

458bool Find::item_text_match (SPItem *item, const gchar *find, bool exact, bool casematch, bool replace/*=false*/)
459{

461 return false;
462 }
463
464 Glib::ustring item_text = sp_te_get_string_multiline(item);
465
466 if (!item_text.empty()) {
467 bool found = find_strcmp(item_text.c_str(), find, exact, casematch);
468
469 if (found && replace) {
470 Glib::ustring ufind = find;
471 if (!casematch) {
472 ufind = ufind.lowercase();
473 }

476 if (!layout) {
477 return found;
478 }

481 gsize n = find_strcmp_pos(item_text.c_str(), ufind.c_str(), exact, casematch);

487 sp_te_replace(item, _begin_w, _end_w, replace.c_str());
488 item_text = sp_te_get_string_multiline (item);
489 n = find_strcmp_pos(item_text.c_str(), ufind.c_str(), exact, casematch, n + replace.length());
490 }
491 }

494 }
