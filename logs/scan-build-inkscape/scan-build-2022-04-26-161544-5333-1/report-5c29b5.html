309/** Insert a new node at the extremum of the selected segments. */
310void PathManipulator::insertNodeAtExtremum(ExtremumType extremum)
311{
312 if (_selection.size() < 2) return;
313
314 double sign = (extremum == EXTR_MIN_X || extremum == EXTR_MIN_Y) ? -1. : 1.;
315 Geom::Dim2 dim = (extremum == EXTR_MIN_X || extremum == EXTR_MAX_X) ? Geom::X : Geom::Y;
316
317 for (auto & _subpath : _subpaths) {
318 Geom::Coord extrvalue = - Geom::infinity();
319 std::vector< std::pair<NodeList::iterator, double> > extremum_vector;
320
321 for (NodeList::iterator first = _subpath->begin(); first != _subpath->end(); ++first) {
322 NodeList::iterator second = first.next();
323 if (second && first->selected() && second->selected()) {
324 add_or_replace_if_extremum(extremum_vector, extrvalue, sign * first->position()[dim], first, 0.);
325 add_or_replace_if_extremum(extremum_vector, extrvalue, sign * second->position()[dim], first, 1.);
326 if (first->front()->isDegenerate() && second->back()->isDegenerate()) {
327 // a line segment has is extrema at the start and end, no node should be added
328 continue;
329 } else {
330 // build 1D cubic bezier curve
331 Geom::Bezier temp1d(first->position()[dim], first->front()->position()[dim],
332 second->back()->position()[dim], second->position()[dim]);
333 // and determine extremum
334 Geom::Bezier deriv1d = derivative(temp1d);
335 std::vector<double> rs = deriv1d.roots();
336 for (double & r : rs) {
337 add_or_replace_if_extremum(extremum_vector, extrvalue, sign * temp1d.valueAt(r), first, r);
338 }
339 }
340 }
341 }
342
343 for (auto & i : extremum_vector) {
344 // don't insert node at the start or end of a segment, i.e. round values for extr_t
345 double t = i.second;
346 if ( !Geom::are_near(t - std::floor(t+0.5),0.) ) // std::floor(t+0.5) is another way of writing round(t)
347 {
348 _selection.insert( subdivideSegment(i.first, t).ptr() );
349 }
350 }
351 }

439/** Replace contiguous selections of nodes in each subpath with one node. */
440void PathManipulator::weldNodes(NodeList::iterator preserve_pos)
441{
442 if (_selection.size() < 2) return;
443 hideDragPoint();
444
445 bool pos_valid = preserve_pos;
446 for (auto sp : _subpaths) {
447 unsigned num_selected = 0, num_unselected = 0;
448 for (auto & j : *sp) {
449 if (j.selected()) ++num_selected;
450 else ++num_unselected;
451 }
452 if (num_selected < 2) continue;
453 if (num_unselected == 0) {
454 // if all nodes in a subpath are selected, the operation doesn't make much sense
455 continue;
456 }
457
458 // Start from unselected node in closed paths, so that we don't start in the middle
459 // of a selection
460 NodeList::iterator sel_beg = sp->begin(), sel_end;
461 if (sp->closed()) {
462 while (sel_beg->selected()) ++sel_beg;
463 }
464
465 // Work loop
466 while (num_selected > 0) {
467 // Find selected node
468 while (sel_beg && !sel_beg->selected()) sel_beg = sel_beg.next();
469 if (!sel_beg) throw std::logic_error("Join nodes: end of open path reached, "
470 "but there are still nodes to process!");
471
472 // note: this is initialized to zero, because the loop below counts sel_beg as well
473 // the loop conditions are simpler that way
474 unsigned num_points = 0;
475 bool use_pos = false;
476 Geom::Point back_pos, front_pos;
477 back_pos = *sel_beg->back();
478
479 for (sel_end = sel_beg; sel_end && sel_end->selected(); sel_end = sel_end.next()) {
480 ++num_points;
481 front_pos = *sel_end->front();
482 if (pos_valid && sel_end == preserve_pos) use_pos = true;
483 }
484 if (num_points > 1) {
485 Geom::Point joined_pos;
486 if (use_pos) {
487 joined_pos = preserve_pos->position();
488 pos_valid = false;
489 } else {
490 joined_pos = Geom::middle_point(back_pos, front_pos);
491 }
492 sel_beg->setType(NODE_CUSP, false);
493 sel_beg->move(joined_pos);
494 // do not move handles if they aren't degenerate
495 if (!sel_beg->back()->isDegenerate()) {
496 sel_beg->back()->setPosition(back_pos);
497 }
498 if (!sel_end.prev()->front()->isDegenerate()) {
499 sel_beg->front()->setPosition(front_pos);

501 sel_beg = sel_beg.next();
502 while (sel_beg != sel_end) {
503 NodeList::iterator next = sel_beg.next();

507 }

525 }
526
527 // if 2 or fewer nodes are selected, there can't be any middle points to remove.
528 if (num_selected <= 2) continue;
529
530 if (num_unselected == 0 && sp->closed()) {
531 // if all nodes in a closed subpath are selected, the operation doesn't make much sense
532 continue;
533 }
534
535 // Start from unselected node in closed paths, so that we don't start in the middle
536 // of a selection
537 NodeList::iterator sel_beg = sp->begin(), sel_end;
538 if (sp->closed()) {
539 while (sel_beg->selected()) ++sel_beg;
540 }
541
542 // Work loop
543 while (num_selected > 0) {
544 // Find selected node
545 while (sel_beg && !sel_beg->selected()) sel_beg = sel_beg.next();
546 if (!sel_beg) throw std::logic_error("Join nodes: end of open path reached, "
547 "but there are still nodes to process!");
548
549 // note: this is initialized to zero, because the loop below counts sel_beg as well
550 // the loop conditions are simpler that way
551 unsigned num_points = 0;
552
553 // find the end of selected segment
554 for (sel_end = sel_beg; sel_end && sel_end->selected(); sel_end = sel_end.next()) {
555 ++num_points;
556 }
557 if (num_points > 2) {
558 // remove nodes in the middle
559 // TODO: fit bezier to the former shape
560 sel_beg = sel_beg.next();
561 while (sel_beg != sel_end.prev()) {
562 NodeList::iterator next = sel_beg.next();
563 sp->erase(sel_beg);
564 sel_beg = next;
565 }
566 }
567 sel_beg = sel_end;
568 // decrease num_selected by the number of points processed
569 num_selected -= num_points;
570 }
571 }
572}

574/** Break the subpath at selected nodes. It also works for single node closed paths. */
575void PathManipulator::breakNodes()
576{
577 for (SubpathList::iterator i = _subpaths.begin(); i != _subpaths.end(); ++i) {
578 SubpathPtr sp = *i;
579 NodeList::iterator cur = sp->begin(), end = sp->end();
580 if (!sp->closed()) {
581 // Each open path must have at least two nodes so no checks are required.
582 // For 2-node open paths, cur == end
583 ++cur;
584 --end;
585 }
586 for (; cur != end; ++cur) {
587 if (!cur->selected()) continue;
588 SubpathPtr ins;
589 bool becomes_open = false;
590
591 if (sp->closed()) {
592 // Move the node to break at to the beginning of path
593 if (cur != sp->begin())
594 sp->splice(sp->begin(), *sp, cur, sp->end());
595 sp->setClosed(false);
596 ins = sp;
597 becomes_open = true;
598 } else {
599 SubpathPtr new_sp(new NodeList(_subpaths));
600 new_sp->splice(new_sp->end(), *sp, sp->begin(), cur);
601 _subpaths.insert(i, new_sp);
602 ins = new_sp;
603 }
604
605 Node *n = new Node(_multi_path_manipulator._path_data.node_data, cur->position());
606 ins->insert(ins->end(), n);
607 cur->setType(NODE_CUSP, false);
608 n->back()->setRelativePos(cur->back()->relativePos());
609 cur->back()->retract();
610 n->sink();
611
612 if (becomes_open) {
613 cur = sp->begin(); // this will be increased to ++sp->begin()
614 end = --sp->end();
615 }
616 }
617 }
618}

620/** Delete selected nodes in the path, optionally substituting deleted segments with bezier curves
621 * in a way that attempts to preserve the original shape of the curve. */
622void PathManipulator::deleteNodes(bool keep_shape)
623{
624 if (_selection.empty()) return;
625 hideDragPoint();
626
627 for (SubpathList::iterator i = _subpaths.begin(); i != _subpaths.end();) {
628 SubpathPtr sp = *i;
629
630 // If there are less than 2 unselected nodes in an open subpath or no unselected nodes
631 // in a closed one, delete entire subpath.
632 unsigned num_unselected = 0, num_selected = 0;
633 for (auto & j : *sp) {
634 if (j.selected()) ++num_selected;
635 else ++num_unselected;
636 }
637 if (num_selected == 0) {
638 ++i;
639 continue;
640 }
641 if (sp->closed() ? (num_unselected < 1) : (num_unselected < 2)) {
642 _subpaths.erase(i++);
643 continue;
644 }
645
646 // In closed paths, start from an unselected node - otherwise we might start in the middle
647 // of a selected stretch and the resulting bezier fit would be suboptimal
648 NodeList::iterator sel_beg = sp->begin(), sel_end;
649 if (sp->closed()) {
650 while (sel_beg->selected()) ++sel_beg;
651 }
652 sel_end = sel_beg;
653
654 while (num_selected > 0) {
655 while (sel_beg && !sel_beg->selected()) {
656 sel_beg = sel_beg.next();
657 }
658 sel_end = sel_beg;
659
660 while (sel_end && sel_end->selected()) {
661 sel_end = sel_end.next();
662 }
663
664 num_selected -= _deleteStretch(sel_beg, sel_end, keep_shape);
665 sel_beg = sel_end;
666 }
667 ++i;
668 }
669}

671/**
672 * Delete nodes between the two iterators.
673 * The given range can cross the beginning of the subpath in closed subpaths.
674 * @param start Beginning of the range to delete
675 * @param end End of the range
676 * @param keep_shape Whether to fit the handles at surrounding nodes to approximate
677 * the shape before deletion
678 * @return Number of deleted nodes
679 */
680unsigned PathManipulator::_deleteStretch(NodeList::iterator start, NodeList::iterator end, bool keep_shape)
681{
682 unsigned const samples_per_segment = 10;
683 double const t_step = 1.0 / samples_per_segment;
684
685 unsigned del_len = 0;
686 for (NodeList::iterator i = start; i != end; i = i.next()) {
687 ++del_len;
688 }
689 if (del_len == 0) return 0;
690
691 // set surrounding node types to cusp if:
692 // 1. keep_shape is on, or
693 // 2. we are deleting at the end or beginning of an open path
694 if ((keep_shape || !end) && start.prev()) start.prev()->setType(NODE_CUSP, false);
695 if ((keep_shape || !start.prev()) && end) end->setType(NODE_CUSP, false);
696
697 if (keep_shape && start.prev() && end) {
698 unsigned num_samples = (del_len + 1) * samples_per_segment + 1;
699 Geom::Point *bezier_data = new Geom::Point[num_samples];
700 Geom::Point result[4];
701 unsigned seg = 0;
702
703 for (NodeList::iterator cur = start.prev(); cur != end; cur = cur.next()) {
704 Geom::CubicBezier bc(*cur, *cur->front(), *cur.next(), *cur.next()->back());
705 for (unsigned s = 0; s < samples_per_segment; ++s) {
706 bezier_data[seg * samples_per_segment + s] = bc.pointAt(t_step * s);
707 }
708 ++seg;
709 }
710 // Fill last point
711 bezier_data[num_samples - 1] = end->position();
712 // Compute replacement bezier curve
713 // TODO the fitting algorithm sucks - rewrite it to be awesome
714 bezier_fit_cubic(result, bezier_data, num_samples, 0.5);
715 delete[] bezier_data;
716
717 start.prev()->front()->setPosition(result[1]);
718 end->back()->setPosition(result[2]);
719 }
720
721 // We can't use nl->erase(start, end), because it would break when the stretch
722 // crosses the beginning of a closed subpath
723 NodeList &nl = start->nodeList();
724 while (start != end) {
725 NodeList::iterator next = start.next();
726 nl.erase(start);
727 start = next;
728 }
729 // if we are removing, we readjust the handlers
730 if(_isBSpline()){
731 if(start.prev()){
732 double bspline_weight = _bsplineHandlePosition(start.prev()->back(), false);
733 start.prev()->front()->setPosition(_bsplineHandleReposition(start.prev()->front(), bspline_weight));
734 }
735 if(end){
736 double bspline_weight = _bsplineHandlePosition(end->front(), false);
737 end->back()->setPosition(_bsplineHandleReposition(end->back(),bspline_weight));
738 }
739 }
740
741 return del_len;
742}

744/** Removes selected segments */
745void PathManipulator::deleteSegments()
746{
747 if (_selection.empty()) return;
748 hideDragPoint();
749
750 for (SubpathList::iterator i = _subpaths.begin(); i != _subpaths.end();) {
751 SubpathPtr sp = *i;
752 bool has_unselected = false;
753 unsigned num_selected = 0;
754 for (auto & j : *sp) {
755 if (j.selected()) {
756 ++num_selected;
757 } else {
758 has_unselected = true;
759 }
760 }
761 if (!has_unselected) {
762 _subpaths.erase(i++);
763 continue;
764 }
765
766 NodeList::iterator sel_beg = sp->begin();
767 if (sp->closed()) {
768 while (sel_beg && sel_beg->selected()) ++sel_beg;
769 }
770 while (num_selected > 0) {
771 if (!sel_beg->selected()) {
772 sel_beg = sel_beg.next();
773 continue;
774 }
775 NodeList::iterator sel_end = sel_beg;
776 unsigned num_points = 0;
777 while (sel_end && sel_end->selected()) {
778 sel_end = sel_end.next();
779 ++num_points;
780 }
781 if (num_points >= 2) {
782 // Retract end handles
783 sel_end.prev()->setType(NODE_CUSP, false);
784 sel_end.prev()->back()->retract();
785 sel_beg->setType(NODE_CUSP, false);
786 sel_beg->front()->retract();
787 if (sp->closed()) {
788 // In closed paths, relocate the beginning of the path to the last selected
789 // node and then unclose it. Remove the nodes from the first selected node
790 // to the new end of path.
791 if (sel_end.prev() != sp->begin())
792 sp->splice(sp->begin(), *sp, sel_end.prev(), sp->end());
793 sp->setClosed(false);
794 sp->erase(sel_beg.next(), sp->end());
795 } else {
796 // for open paths:
797 // 1. At end or beginning, delete including the node on the end or beginning
798 // 2. In the middle, delete only inner nodes
799 if (sel_beg == sp->begin()) {
800 sp->erase(sp->begin(), sel_end.prev());
801 } else if (sel_end == sp->end()) {
802 sp->erase(sel_beg.next(), sp->end());
803 } else {
804 SubpathPtr new_sp(new NodeList(_subpaths));
805 new_sp->splice(new_sp->end(), *sp, sp->begin(), sel_beg.next());
806 _subpaths.insert(i, new_sp);
807 if (sel_end.prev())
808 sp->erase(sp->begin(), sel_end.prev());
809 }
810 }
811 }
812 sel_beg = sel_end;
813 num_selected -= num_points;
814 }
815 ++i;
816 }
817}

838/** Make selected segments curves / lines. */
839void PathManipulator::setSegmentType(SegmentType type)
840{
841 if (_selection.empty()) return;
842 for (auto & _subpath : _subpaths) {
843 for (NodeList::iterator j = _subpath->begin(); j != _subpath->end(); ++j) {
844 NodeList::iterator k = j.next();
845 if (!(k && j->selected() && k->selected())) continue;
846 switch (type) {

855 break;

857 j->front()->move(j->position() + (k->position() - j->position()) / 3);
858 k->back()->move(k->position() + (j->position() - k->position()) / 3);
859 break;
860 }
861 }

922Handle *PathManipulator::_chooseHandle(Node *n, int which)
923{
924 NodeList::iterator i = NodeList::get_iterator(n);
925 Node *prev = i.prev().ptr();
926 Node *next = i.next().ptr();
927
928 // on an endnode, the remaining handle automatically wins
929 if (!next) return n->back();
930 if (!prev) return n->front();
931
932 // compare X coord offline segments
933 Geom::Point npos = next->position();
934 Geom::Point ppos = prev->position();
935 if (which < 0) {
936 // pick left handle.
937 // we just swap the handles and pick the right handle below.
938 std::swap(npos, ppos);
939 }
940
941 if (npos[Geom::X] >= ppos[Geom::X]) {
942 return n->front();
943 } else {
944 return n->back();
945 }
946}

1026/** Insert a node in the segment beginning with the supplied iterator,
1027 * at the given time value */
1028NodeList::iterator PathManipulator::subdivideSegment(NodeList::iterator first, double t)
1029{
1030 if (!first) throw std::invalid_argument("Subdivide after invalid iterator");
1031 NodeList &list = NodeList::get(first);
1032 NodeList::iterator second = first.next();
1033 if (!second) throw std::invalid_argument("Subdivide after last node in open path");
1034 if (first->type() == NODE_SYMMETRIC)
1035 first->setType(NODE_SMOOTH, false);
1036 if (second->type() == NODE_SYMMETRIC)
1037 second->setType(NODE_SMOOTH, false);
1038
1039 // We need to insert the segment after 'first'. We can't simply use 'second'
1040 // as the point of insertion, because when 'first' is the last node of closed path,
1041 // the new node will be inserted as the first node instead.
1042 NodeList::iterator insert_at = first;
1043 ++insert_at;
1044
1045 NodeList::iterator inserted;
1046 if (first->front()->isDegenerate() && second->back()->isDegenerate()) {
1047 // for a line segment, insert a cusp node
1048 Node *n = new Node(_multi_path_manipulator._path_data.node_data,
1049 Geom::lerp(t, first->position(), second->position()));
1050 n->setType(NODE_CUSP, false);
1051 inserted = list.insert(insert_at, n);
1052 } else {
1053 // build bezier curve and subdivide
1054 Geom::CubicBezier temp(first->position(), first->front()->position(),
1055 second->back()->position(), second->position());
1056 std::pair<Geom::CubicBezier, Geom::CubicBezier> div = temp.subdivide(t);
1057 std::vector<Geom::Point> seg1 = div.first.controlPoints(), seg2 = div.second.controlPoints();
1058
1059 // set new handle positions
1060 Node *n = new Node(_multi_path_manipulator._path_data.node_data, seg2[0]);
1061 if(!_isBSpline()){
1062 n->back()->setPosition(seg1[2]);
1063 n->front()->setPosition(seg2[1]);
1064 n->setType(NODE_SMOOTH, false);
1065 } else {
1066 Geom::D2< Geom::SBasis > sbasis_inside_nodes;
1067 std::unique_ptr<SPCurve> line_inside_nodes(new SPCurve());
1068 if(second->back()->isDegenerate()){
1069 line_inside_nodes->moveto(n->position());
1070 line_inside_nodes->lineto(second->position());
1071 sbasis_inside_nodes = line_inside_nodes->first_segment()->toSBasis();
1072 Geom::Point next = sbasis_inside_nodes.valueAt(DEFAULT_START_POWER);
1073 next = Geom::Point(next[Geom::X] + HANDLE_CUBIC_GAP,next[Geom::Y] + HANDLE_CUBIC_GAP);
1074 line_inside_nodes->reset();
1075 n->front()->setPosition(next);
1076 }else{
1077 n->front()->setPosition(seg2[1]);
1078 }
1079 if(first->front()->isDegenerate()){
1080 line_inside_nodes->moveto(n->position());
1081 line_inside_nodes->lineto(first->position());
1082 sbasis_inside_nodes = line_inside_nodes->first_segment()->toSBasis();
1083 Geom::Point previous = sbasis_inside_nodes.valueAt(DEFAULT_START_POWER);
1084 previous = Geom::Point(previous[Geom::X] + HANDLE_CUBIC_GAP,previous[Geom::Y] + HANDLE_CUBIC_GAP);
1085 n->back()->setPosition(previous);
1086 }else{
1087 n->back()->setPosition(seg1[2]);
1088 }
1089 n->setType(NODE_CUSP, false);
1090 }
1091 inserted = list.insert(insert_at, n);
1092
1093 first->front()->move(seg1[1]);
1094 second->back()->move(seg2[2]);
1095 }
1096 return inserted;
1097}

1099/** Find the node that is closest/farthest from the origin
1100 * @param origin Point of reference
1101 * @param search_selected Consider selected nodes
1102 * @param search_unselected Consider unselected nodes
1103 * @param closest If true, return closest node, if false, return farthest
1104 * @return The matching node, or an empty iterator if none found
1105 */
1106NodeList::iterator PathManipulator::extremeNode(NodeList::iterator origin, bool search_selected,
1107 bool search_unselected, bool closest)
1108{
1109 NodeList::iterator match;
1110 double extr_dist = closest ? HUGE_VAL(__builtin_huge_val ()) : -HUGE_VAL(__builtin_huge_val ());
1111 if (_selection.empty() && !search_unselected) return match;
1112
1113 for (auto & _subpath : _subpaths) {
1114 for (NodeList::iterator j = _subpath->begin(); j != _subpath->end(); ++j) {
1115 if(j->selected()) {
1116 if (!search_selected) continue;
1117 } else {
1118 if (!search_unselected) continue;
1119 }
1120 double dist = Geom::distance(*j, *origin);
1121 bool cond = closest ? (dist < extr_dist) : (dist > extr_dist);
1122 if (cond) {
1123 match = j;
1124 extr_dist = dist;
1125 }
1126 }
1127 }
1128 return match;
1129}

1188/** Create nodes and handles based on the XML of the edited path. */
1189void PathManipulator::_createControlPointsFromGeometry()
1190{
1191 clear();
1192
1193 // sanitize pathvector and store it in SPCurve,
1194 // so that _updateDragPoint doesn't crash on paths with naked movetos
1195 Geom::PathVector pathv = pathv_to_linear_and_cubic_beziers(_spcurve->get_pathvector());
1196 for (Geom::PathVector::iterator i = pathv.begin(); i != pathv.end(); ) {
1197 // NOTE: this utilizes the fact that Geom::PathVector is an std::vector.
1198 // When we erase an element, the next one slides into position,
1199 // so we do not increment the iterator even though it is theoretically invalidated.
1200 if (i->empty()) {
1201 i = pathv.erase(i);
1202 } else {
1203 ++i;
1204 }
1205 }
1206 if (pathv.empty()) {
1207 return;
1208 }
1209 _spcurve->set_pathvector(pathv);
1210
1211 pathv *= (_edit_transform * _i2d_transform);
1212
1213 // in this loop, we know that there are no zero-segment subpaths
1214 for (auto & pit : pathv) {
1215 // prepare new subpath
1216 SubpathPtr subpath(new NodeList(_subpaths));
1217 _subpaths.push_back(subpath);
1218
1219 Node *previous_node = new Node(_multi_path_manipulator._path_data.node_data, pit.initialPoint());
1220 subpath->push_back(previous_node);
1221
1222 bool closed = pit.closed();
1223
1224 for (Geom::Path::iterator cit = pit.begin(); cit != pit.end(); ++cit) {
1225 Geom::Point pos = cit->finalPoint();
1226 Node *current_node;
1227 // if the closing segment is degenerate and the path is closed, we need to move
1228 // the handle of the first node instead of creating a new one
1229 if (closed && cit == --(pit.end())) {
1230 current_node = subpath->begin().get_pointer();
1231 } else {
1232 /* regardless of segment type, create a new node at the end
1233 * of this segment (unless this is the last segment of a closed path
1234 * with a degenerate closing segment */
1235 current_node = new Node(_multi_path_manipulator._path_data.node_data, pos);
1236 subpath->push_back(current_node);
1237 }
1238 // if this is a bezier segment, move handles appropriately
1239 // TODO: I don't know why the dynamic cast below doesn't want to work
1240 // when I replace BezierCurve with CubicBezier. Might be a bug
1241 // somewhere in pathv_to_linear_and_cubic_beziers
1242 Geom::BezierCurve const *bezier = dynamic_cast<Geom::BezierCurve const*>(&*cit);
1243 if (bezier && bezier->order() == 3)
1244 {
1245 previous_node->front()->setPosition((*bezier)[1]);
1246 current_node ->back() ->setPosition((*bezier)[2]);
1247 }
1248 previous_node = current_node;
1249 }
1250 // If the path is closed, make the list cyclic
1251 if (pit.closed()) subpath->setClosed(true);
1252 }
1253
1254 // we need to set the nodetypes after all the handles are in place,
1255 // so that pickBestType works correctly
1256 // TODO maybe migrate to inkscape:node-types?
1257 // TODO move this into SPPath - do not manipulate directly
1258
1259 //XML Tree being used here directly while it shouldn't be.
1260 gchar const *nts_raw = _path ? _path->getRepr()->attribute(_nodetypesKey().data()) : nullptr;
1261 /* Calculate the needed length of the nodetype string.
1262 * For closed paths, the entry is duplicated for the starting node,
1263 * so we can just use the count of segments including the closing one
1264 * to include the extra end node. */
1265 /* pad the string to required length with a bogus value.
1266 * 'b' and any other letter not recognized by the parser causes the best fit to be set
1267 * as the node type */
1268 auto const *tsi = nts_raw ? nts_raw : "";
1269 for (auto & _subpath : _subpaths) {
1270 for (auto & j : *_subpath) {
1271 char nodetype = (*tsi) ? (*tsi++) : 'b';
1272 j.setType(Node::parse_nodetype(nodetype), false);
1273 }
1274 if (_subpath->closed() && *tsi) {
1275 // STUPIDITY ALERT: it seems we need to use the duplicate type symbol instead of
1276 // the first one to remain backward compatible.
1277 _subpath->begin()->setType(Node::parse_nodetype(*tsi++), false);
1278 }
1279 }

1483/** Update the path outline. */
1484void PathManipulator::_updateOutline()
1485{
1486 if (!_show_outline) {
1487 _outline->hide();
1488 return;
1489 }

1491 Geom::PathVector pv = _spcurve->get_pathvector();

1730/** Update the position of the curve drag point such that it is over the nearest
1731 * point of the path. */
1732Geom::Coord PathManipulator::_updateDragPoint(Geom::Point const &evp)
1733{
1734 Geom::Coord dist = HUGE_VAL(__builtin_huge_val ());
1735
1736 Geom::Affine to_desktop = _edit_transform * _i2d_transform;
1737 Geom::PathVector pv = _spcurve->get_pathvector();
1738 std::optional<Geom::PathVectorTime> pvp =
1739 pv.nearestTime(_desktop->w2d(evp) * to_desktop.inverse());
1740 if (!pvp) return dist;
1741 Geom::Point nearest_pt = _desktop->d2w(pv.pointAt(*pvp) * to_desktop);
1742
1743 double fracpart = pvp->t;
1744 std::list<SubpathPtr>::iterator spi = _subpaths.begin();
1745 for (unsigned i = 0; i < pvp->path_index; ++i, ++spi) {}
1746 NodeList::iterator first = (*spi)->before(pvp->asPathTime());
1747
1748 dist = Geom::distance(evp, nearest_pt);
1749
1750 double stroke_tolerance = _getStrokeTolerance();
1751 if (first && first.next() &&
1752 fracpart != 0.0 &&
1753 fracpart != 1.0 &&
1754 dist < stroke_tolerance)
1755 {
1756 // stroke_tolerance is at least two.
1757 int tolerance = std::max(2, (int)stroke_tolerance);
1758 _dragpoint->setVisible(true);
1759 _dragpoint->setPosition(_desktop->w2d(nearest_pt));
1760 _dragpoint->setSize(2 * tolerance - 1);
1761 _dragpoint->setTimeValue(fracpart);
1762 _dragpoint->setIterator(first);
1763 } else {
1764 _dragpoint->setVisible(false);
1765 }
1766
1767 return dist;
1768}

1776/** Compute the radius from the edge of the path where clicks should initiate a curve drag
1777 * or segment selection, in window coordinates. */
1778double PathManipulator::_getStrokeTolerance()
1779{

1781 * drag tolerance setting. */
1782 Inkscape::Preferences *prefs = Inkscape::Preferences::get();
1783 double ret = prefs->getIntLimited("/options/dragtolerance/value", 2, 0, 100);
