50/** @brief Compute the value of a Bernstein-Bezier polynomial.
51 * This method uses a Horner-like fast evaluation scheme.
52 * @param t Time value
53 * @param c_ Pointer to coefficients
54 * @param n Degree of the polynomial (number of coefficients minus one) */
55template <typename T>
56inline T bernstein_value_at(double t, T const *c_, unsigned n) {
57 double u = 1.0 - t;
58 double bc = 1;
59 double tn = 1;
60 T tmp = c_[0]*u;
61 for(unsigned i=1; i<n; i++){
62 tn = tn*t;
63 bc = bc*(n-i+1)/i;
64 tmp = (tmp + tn*bc*c_[i])*u;
65 }
66 return (tmp + tn*t*c_[n]);
67}
