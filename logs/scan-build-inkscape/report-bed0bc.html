506 /* Whole text's style */
507 Glib::ustring style_str =
508 item->style->writeIfDiff(item->parent ? item->parent->style : nullptr); // TODO investigate possibility
509 g_repr->setAttributeOrRemoveIfEmpty("style", style_str);
510
511 Inkscape::Text::Layout::iterator iter = te_get_layout(item)->begin();
512 do {
513 Inkscape::Text::Layout::iterator iter_next = iter;
514 iter_next.nextGlyph(); // iter_next is one glyph ahead from iter
515 if (iter == iter_next)
516 break;
517
518 /* This glyph's style */
519 SPObject *pos_obj = nullptr;
520 te_get_layout(item)->getSourceOfCharacter(iter, &pos_obj);
521 if (!pos_obj) // no source for glyph, abort
522 break;
523 while (dynamic_cast<SPString const *>(pos_obj) && pos_obj->parent) {
524 pos_obj = pos_obj->parent; // SPStrings don't have style
525 }
526 Glib::ustring style_str = pos_obj->style->writeIfDiff(item->style);
527
528 // get path from iter to iter_next:
529 auto curve = te_get_layout(item)->convertToCurves(iter, iter_next);
530 iter = iter_next; // shift to next glyph
531 if (!curve) { // error converting this glyph
532 continue;
533 }
534 if (curve->is_empty()) { // whitespace glyph?
535 continue;
536 }
537
538 Inkscape::XML::Node *p_repr = xml_doc->createElement("svg:path");
539
540 p_repr->setAttribute("d", sp_svg_write_path(curve->get_pathvector()));
541
542 p_repr->setAttributeOrRemoveIfEmpty("style", style_str);
543
544 g_repr->appendChild(p_repr);
545
546 Inkscape::GC::release(p_repr);
547
548 if (iter == te_get_layout(item)->end())
549 break;
550
551 } while (true);
552
553 return g_repr;
554 }
555
556 std::unique_ptr<SPCurve> curve;
557
558 {
559 SPShape *shape = dynamic_cast<SPShape *>(item);
560 if (shape) {
561 curve = SPCurve::copy(shape->curveForEdit());
562 }
563 }
564
565 if (!curve)
566 return nullptr;
567
568 // Prevent empty paths from being added to the document
569 // otherwise we end up with zomby markup in the SVG file
570 if(curve->is_empty())
571 {
572 return nullptr;
573 }
574
575 Inkscape::XML::Node *repr = xml_doc->createElement("svg:path");
576
577 Inkscape::copy_object_properties(repr, item->getRepr());
578
579 /* Transformation */
580 repr->setAttribute("transform", item->getRepr()->attribute("transform"));
581
582 /* Style */
583 Glib::ustring style_str =
584 item->style->writeIfDiff(item->parent ? item->parent->style : nullptr); // TODO investigate possibility
585 repr->setAttributeOrRemoveIfEmpty("style", style_str);
586
587 /* Definition */
588 repr->setAttribute("d", sp_svg_write_path(curve->get_pathvector()));
589 return repr;
590}

593void
594ObjectSet::pathReverse()
595{
596 if (isEmpty()) {
597 if(desktop())
598 desktop()->getMessageStack()->flash(Inkscape::WARNING_MESSAGE, _("Select <b>path(s)</b> to reverse.")gettext ("Select <b>path(s)</b> to reverse."));
599 return;
600 }
601
602
603 // set "busy" cursor
604 if(desktop()){
605 desktop()->setWaitingCursor();
606 desktop()->messageStack()->flash(Inkscape::IMMEDIATE_MESSAGE, _("Reversing paths...")gettext ("Reversing paths..."));
607 }
608
609 bool did = false;
610
611 for (auto i = items().begin(); i != items().end(); ++i){
612
613 SPPath *path = dynamic_cast<SPPath *>(*i);
614 if (!path) {
615 continue;
616 }
617
618 did = true;
619
620 auto rcurve = path->curveForEdit()->create_reverse();
621
622 auto str = sp_svg_write_path(rcurve->get_pathvector());
623 if ( path->hasPathEffectRecursive() ) {
624 path->setAttribute("inkscape:original-d", str);
625 } else {
626 path->setAttribute("d", str);
627 }
628
629 // reverse nodetypes order (Bug #179866)
630 gchar *nodetypes = g_strdup(path->getRepr()->attribute("sodipodi:nodetypes"));
631 if ( nodetypes ) {
632 path->setAttribute("sodipodi:nodetypes", g_strreverse(nodetypes));
633 g_free(nodetypes);
634 }
635 }
636 if(desktop())
637 desktop()->clearWaitingCursor();
638
639 if (did) {
640 DocumentUndo::done(document(), _("Reverse path")gettext ("Reverse path"), INKSCAPE_ICON("path-reverse")"path-reverse");
641 } else {
642 if(desktop())
643 desktop()->getMessageStack()->flash(Inkscape::ERROR_MESSAGE, _("<b>No paths</b> to reverse in the selection.")gettext ("<b>No paths</b> to reverse in the selection."
));
644 }
645}

648/**
649 * Copy generic attributes, like those from the "Object Properties" dialog,
650 * but also style and transformation center.
651 *
652 * @param dest XML node to copy attributes to
653 * @param src XML node to copy attributes from
654 */
655static void ink_copy_generic_attributes( //
656 Inkscape::XML::Node *dest, //
657 Inkscape::XML::Node const *src)
658{
659 static char const *const keys[] = {
660 // core
661 "id",
662
663 // clip & mask
664 "clip-path",
665 "mask",
666
667 // style
668 "style",
669 "class",
670
671 // inkscape
672 "inkscape:highlight-color",
673 "inkscape:label",
674 "inkscape:transform-center-x",
675 "inkscape:transform-center-y",
676
677 // interactivity
678 "onclick",
679 "onmouseover",
680 "onmouseout",
681 "onmousedown",
682 "onmouseup",
683 "onmousemove",
684 "onfocusin",
685 "onfocusout",
686 "onload",
687 };
688
689 for (auto *key : keys) {
690 auto *value = src->attribute(key);
691 if (value) {
692 dest->setAttribute(key, value);
693 }
694 }
695}

698/**
699 * Copy generic child elements, like those from the "Object Properties" dialog
700 * (title and description) but also XML comments.
701 *
702 * Does not check if children of the same type already exist in dest.
703 *
704 * @param dest XML node to copy children to
705 * @param src XML node to copy children from
706 */
707static void ink_copy_generic_children( //
708 Inkscape::XML::Node *dest, //
709 Inkscape::XML::Node const *src)
710{
711 static std::set<std::string> const names{
712 // descriptive elements
713 "svg:title",
714 "svg:desc",
715 };
716
717 for (const auto *child = src->firstChild(); child != nullptr; child = child->next()) {
718 // check if this child should be copied
719 if (!(child->type() == Inkscape::XML::NodeType::COMMENT_NODE || //
720 (child->name() && names.count(child->name())))) {
721 continue;
722 }
723
724 auto dchild = child->duplicate(dest->document());
725 dest->appendChild(dchild);
726 dchild->release();
727 }
728}

731/**
732 * Copy generic object properties, like:
733 * - id
734 * - label
735 * - title
736 * - description
737 * - style
738 * - clip
739 * - mask
740 * - transformation center
741 * - highlight color
742 * - interactivity (event attributes)
743 *
744 * @param dest XML node to copy to
745 * @param src XML node to copy from
746 */
747void Inkscape::copy_object_properties( //
748 Inkscape::XML::Node *dest, //
749 Inkscape::XML::Node const *src)
750{
751 ink_copy_generic_attributes(dest, src);
752 ink_copy_generic_children(dest, src);
753}
