1183 pos = pos - Point::polar(angle_cross, position + (anotation_width/2.0));
1184 angle += Geom::rad_from_deg(90);
1185 } else {
1186 pos = pos - Point::polar(angle_cross, position);

1188 }
1189 if (!scale_sensitive && !parentaffinetransform.preservesAngles()) {
1190 createTextLabel(pos, counter, length, angle, remove, false);

1195 if(flip_side) {
1196 arrow_gap *= -1;
1197 }
1198 if(hide_arrows) {
1199 arrow_gap *= 0;
1200 }
1201 createLine(end, hend, Glib::ustring("infoline-on-end-"), counter, false, false);
1202 if (!arrows_outside) {
1203 hstart = hstart + Point::polar(angle, arrow_gap);
1204 hend = hend - Point::polar(angle, arrow_gap );
1205 }
1206 if ((Geom::distance(hstart, hend) / 2.0) > (anotation_width / 1.9) + arrow_gap) {

1208 } else {
1209 createLine(hstart, hend, Glib::ustring("infoline-"), counter, true, true, true);
1210 }
1211 } else {
1212 createLine(Geom::Point(), Geom::Point(), Glib::ustring("infoline-"), counter, true, true, true);
1213 createLine(Geom::Point(), Geom::Point(), Glib::ustring("infoline-on-start-"), counter, true, true, true);
1214 createLine(Geom::Point(), Geom::Point(), Glib::ustring("infoline-on-end-"), counter, true, true, true);
1215 createTextLabel(Geom::Point(), counter, 0, 0, true, true);
1216 }
1217 }
1218 }
1219 if (previous_size) {
1220 for (size_t counter = ncurves; counter < previous_size; counter++) {
1221 createLine(Geom::Point(), Geom::Point(), Glib::ustring("infoline-"), counter, true, true, true);
1222 createLine(Geom::Point(), Geom::Point(), Glib::ustring("infoline-on-start-"), counter, true, true, true);
1223 createLine(Geom::Point(), Geom::Point(), Glib::ustring("infoline-on-end-"), counter, true, true, true);
1224 createTextLabel(Geom::Point(), counter, 0, 0, true, true);
1225 }
1226 }
1227 previous_size = ncurves;
1228 }

1279 elemnode->removeAttribute("sodipodi:insensitive");

1283 }

1286 case LPE_ERASE:
1287 item->deleteObject(true);
1288 break;

1290 case LPE_VISIBILITY:

1300 sp_repr_css_attr_unref (css);

1303 default:
1304 break;
1305 }

1309 items.clear();
