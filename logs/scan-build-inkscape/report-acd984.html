1077
1078 // Find the scale factors for the geometric bbox

1080 Geom::Scale geom_scale = calcScaleFactors(_point_geom, pt_geom, _origin_for_specpoints);
1081
1082 _absolute_affine = Geom::identity(); //Initialize the scaler
1083
1084 auto increments = Modifiers::Modifier::get(Modifiers::Type::TRANS_INCREMENT)->active(state);
1085 if (increments) { // scale by an integer multiplier/divider
1086 // We're scaling either the visual or the geometric bbox here (see the comment above)
1087 for ( unsigned int i = 0 ; i < 2 ; i++ ) {
1088 if (fabs(default_scale[i]) > 1) {
1089 default_scale[i] = round(default_scale[i]);
1090 } else if (default_scale[i] != 0) {
1091 default_scale[i] = 1/round(1/(MIN(default_scale[i], 10)(((default_scale[i]) < (10)) ? (default_scale[i]) : (10))));
1092 }
1093 }
1094 // Update the knot position
1095 pt = _calcAbsAffineDefault(default_scale);
1096 // When scaling by an integer, snapping is not needed
1097 } else {
1098 // In all other cases we should try to snap now

1100
1101 auto confine = Modifiers::Modifier::get(Modifiers::Type::TRANS_CONFINE)->active(state);
1102 if (confine || _desktop->isToolboxButtonActive ("lock")) {
1103 // Scale is locked to a 1:1 aspect ratio, so that s[X] must be made to equal s[Y].
1104 //
1105 // The aspect-ratio must be locked before snapping
1106 if (fabs(default_scale[Geom::X]) > fabs(default_scale[Geom::Y])) {
1107 default_scale[Geom::X] = fabs(default_scale[Geom::Y]) * sign(default_scale[Geom::X]);
1108 geom_scale[Geom::X] = fabs(geom_scale[Geom::Y]) * sign(geom_scale[Geom::X]);

1110 default_scale[Geom::Y] = fabs(default_scale[Geom::X]) * sign(default_scale[Geom::Y]);
1111 geom_scale[Geom::Y] = fabs(geom_scale[Geom::X]) * sign(geom_scale[Geom::Y]);
1112 }

1115
1116 bb = new Inkscape::PureScaleConstrained(default_scale, _origin_for_bboxpoints);
1117 sn = new Inkscape::PureScaleConstrained(geom_scale, _origin_for_specpoints);
1118 } else {
1119 /* Scale aspect ratio is unlocked */
1120 bb = new Inkscape::PureScale(default_scale, _origin_for_bboxpoints, false);
1121 sn = new Inkscape::PureScale(geom_scale, _origin_for_specpoints, false);
1122 }
1123 SnapManager &m = _desktop->namedview->snap_manager;
1124 m.setup(_desktop, false, _objects_const);
1125 m.snapTransformed(_bbox_points, _point, (*bb));
1126 m.snapTransformed(_snap_points, _point, (*sn));
1127 m.unSetup();
1128
1129 // These lines below are duplicated in stretchRequest
1130 //TODO: Eliminate this code duplication
1131 if (bb->best_snapped_point.getSnapped() || sn->best_snapped_point.getSnapped()) {
1132 if (bb->best_snapped_point.getSnapped() && !bb->best_snapped_point.isOtherSnapBetter(sn->best_snapped_point, false)) {
1133 // We snapped the bbox (which is either visual or geometric)
1134 _desktop->snapindicator->set_new_snaptarget(bb->best_snapped_point);
1135 default_scale = bb->getScaleSnapped();
1136 // Calculate the new transformation and update the handle position
1137 pt = _calcAbsAffineDefault(default_scale);
1138 } else if (sn->best_snapped_point.getSnapped()) {
1139 _desktop->snapindicator->set_new_snaptarget(sn->best_snapped_point);
1140 // We snapped the special points (e.g. nodes), which are not at the visual bbox
1141 // The handle location however (pt) might however be at the visual bbox, so we
1142 // will have to calculate pt taking the stroke width into account
1143 geom_scale = sn->getScaleSnapped();

1145 }
1146 } else {
1147 // We didn't snap at all! Don't update the handle position, just calculate the new transformation
1148 _calcAbsAffineDefault(default_scale);
1149 _desktop->snapindicator->remove_snaptarget();
1150 }
1151
1152 delete bb;
1153 delete sn;
1154 }

),
1160 100 * _absolute_affine[0], 100 * _absolute_affine[3], confine_mod.c_str());
1161

1290gboolean Inkscape::SelTrans::skewRequest(SPSelTransHandle const &handle, Geom::Point &pt, guint state)
1291{
1292 /* When skewing (or rotating):
1293 * 1) the stroke width will not change. This makes life much easier because we don't have to
1294 * account for that (like for scaling or stretching). As a consequence, all points will
1295 * have the same origin for the transformation and for the snapping.
1296 * 2) When holding shift, the transformation will be relative to the point opposite of
1297 * the handle; otherwise it will be relative to the center as set for the selection
1298 */
1299
1300 Geom::Dim2 dim_a;
1301 Geom::Dim2 dim_b;
1302
1303 switch (handle.cursor) {
1304 case GDK_SB_H_DOUBLE_ARROW:
1305 dim_a = Geom::Y;
1306 dim_b = Geom::X;
1307 break;
1308 case GDK_SB_V_DOUBLE_ARROW:
1309 dim_a = Geom::X;
1310 dim_b = Geom::Y;
1311 break;
1312 default:
1313 g_assert_not_reached()do { (void) 0; __builtin_unreachable (); } while (0);
1314 abort();
1315 break;
1316 }
1317
1318 // _point and _origin are noisy, ranging from 1 to 1e-9 or even smaller; this is due to the
1319 // limited SVG output precision, which can be arbitrarily set in the preferences
1320 Geom::Point const initial_delta = _point - _origin;
1321
1322 // The handle and the origin shouldn't be too close to each other; let's check for that!
1323 // Due to the limited resolution though (see above), we'd better use a relative error here
1324 if (_bbox) {
1325 Geom::Coord d = (*_bbox).dimensions()[dim_a];
1326 if (fabs(initial_delta[dim_a]/d) < 1e-4) {
1327 return false;
1328 }
1329 }
1330
1331 // Calculate the scale factors, which can be either visual or geometric
1332 // depending on which type of bbox is currently being used (see preferences -> selector tool)
1333 Geom::Scale scale = calcScaleFactors(_point, pt, _origin, false);
1334 Geom::Scale skew = calcScaleFactors(_point, pt, _origin, true);
1335 scale[dim_b] = 1;
1336 skew[dim_b] = 1;
1337
1338 if (fabs(scale[dim_a]) < 1) {
1339 // Prevent shrinking of the selected object, while allowing mirroring
1340 scale[dim_a] = sign(scale[dim_a]);
1341 } else {
1342 // Allow expanding of the selected object by integer multiples
1343 scale[dim_a] = floor(scale[dim_a] + 0.5);
1344 }
1345
1346 double radians = atan(skew[dim_a] / scale[dim_a]);
1347
1348 auto increments = Modifiers::Modifier::get(Modifiers::Type::TRANS_INCREMENT)->active(state);
1349 if (increments) {
1350 Inkscape::Preferences *prefs = Inkscape::Preferences::get();
1351 // Snap to defined angle increments
1352 int snaps = prefs->getInt("/options/rotationsnapsperpi/value", 12);
1353 if (snaps) {
1354 double sections = floor(radians * snaps / M_PI3.14159265358979323846 + .5);
1355 if (fabs(sections) >= snaps / 2) {
1356 sections = sign(sections) * (snaps / 2 - 1);
1357 }
1358 radians = (M_PI3.14159265358979323846 / snaps) * sections;
1359 }
1360 skew[dim_a] = tan(radians) * scale[dim_a];
1361 } else {
1362 // Snap to objects, grids, guides
1363
1364 SnapManager &m = _desktop->namedview->snap_manager;
1365 m.setup(_desktop, false, _objects_const);
1366
1367 // When skewing, we cannot snap the corners of the bounding box, see the comment in PureSkewConstrained for details
1368 Inkscape::PureSkewConstrained sn = Inkscape::PureSkewConstrained(skew[dim_a], scale[dim_a], _origin, Geom::Dim2(dim_b));
1369 m.snapTransformed(_snap_points, _point, sn);
1370
1371 if (sn.best_snapped_point.getSnapped()) {
1372 // We snapped something, so change the skew to reflect it
1373 skew[dim_a] = sn.getSkewSnapped();
1374 _desktop->snapindicator->set_new_snaptarget(sn.best_snapped_point);
1375 } else {
1376 _desktop->snapindicator->remove_snaptarget();
1377 }
1378
1379 m.unSetup();
1380 }
1381
1382 // Update the handle position
1383 pt[dim_b] = initial_delta[dim_a] * skew[dim_a] + _point[dim_b];
1384 pt[dim_a] = initial_delta[dim_a] * scale[dim_a] + _origin[dim_a];
1385
1386 // Calculate the relative affine
1387 _relative_affine = Geom::identity();
1388 _relative_affine[2*dim_a + dim_a] = (pt[dim_a] - _origin[dim_a]) / initial_delta[dim_a];
1389 _relative_affine[2*dim_a + (dim_b)] = (pt[dim_b] - _point[dim_b]) / initial_delta[dim_a];
1390 _relative_affine[2*(dim_b) + (dim_a)] = 0;
1391 _relative_affine[2*(dim_b) + (dim_b)] = 1;
1392
1393 for (int i = 0; i < 2; i++) {
1394 if (fabs(_relative_affine[3*i]) < 1e-15) {
1395 _relative_affine[3*i] = 1e-15;
1396 }
1397 }
1398
1399 // Update the status text
1400 auto increment_mod = Modifiers::Modifier::get(Modifiers::Type::TRANS_INCREMENT)->get_label();
1401 double degrees = mod360symm(Geom::deg_from_rad(radians));
1402 _message_context.setF(Inkscape::IMMEDIATE_MESSAGE,
1403 // TRANSLATORS: don't modify the first ";"
1404 // (it will NOT be displayed as ";" - only the second one will be)
1405 _("<b>Skew</b>: %0.2f&#176;; with <b>%s</b> to snap angle")gettext ("<b>Skew</b>: %0.2f&#176;; with <b>%s</b> to snap angle"
),
1406 degrees, increment_mod.c_str());
1407
1408 return TRUE(!(0));
1409}

1411gboolean Inkscape::SelTrans::rotateRequest(Geom::Point &pt, guint state)
1412{
1413 /* When rotating (or skewing):
1414 * 1) the stroke width will not change. This makes life much easier because we don't have to
1415 * account for that (like for scaling or stretching). As a consequence, all points will
1416 * have the same origin for the transformation and for the snapping.
1417 * 2) When holding shift, the transformation will be relative to the point opposite of
1418 * the handle; otherwise it will be relative to the center as set for the selection
1419 */
1420
1421 Inkscape::Preferences *prefs = Inkscape::Preferences::get();
1422 int snaps = prefs->getInt("/options/rotationsnapsperpi/value", 12);
1423
1424 // rotate affine in rotate
1425 Geom::Point const d1 = _point - _origin;
1426 Geom::Point const d2 = pt - _origin;
1427
1428 Geom::Coord const h1 = Geom::L2(d1); // initial radius
1429 if (h1 < 1e-15) return FALSE(0);
1430 Geom::Point q1 = d1 / h1; // normalized initial vector to handle
1431 Geom::Coord const h2 = Geom::L2(d2); // new radius
1432 if (fabs(h2) < 1e-15) return FALSE(0);
1433 Geom::Point q2 = d2 / h2; // normalized new vector to handle
1434
1435 Geom::Rotate r1(q1);
1436 Geom::Rotate r2(q2);
1437
1438 double radians = atan2(Geom::dot(Geom::rot90(d1), d2), Geom::dot(d1, d2));;
1439 auto increments = Modifiers::Modifier::get(Modifiers::Type::TRANS_INCREMENT)->active(state);
1440 auto confine = Modifiers::Modifier::get(Modifiers::Type::TRANS_CONFINE)->active(state);
1441 // Either key will now snap the rotation to specific points
1442 if (increments || confine) {
1443 // Snap to defined angle increments
1444 double cos_t = Geom::dot(q1, q2);
1445 double sin_t = Geom::dot(Geom::rot90(q1), q2);
1446 radians = atan2(sin_t, cos_t);
1447 if (snaps) {
1448 radians = ( M_PI3.14159265358979323846 / snaps ) * floor( radians * snaps / M_PI3.14159265358979323846 + .5 );
1449 }
1450 r1 = Geom::Rotate(0); //q1 = Geom::Point(1, 0);
1451 r2 = Geom::Rotate(radians); //q2 = Geom::Point(cos(radians), sin(radians));
1452 } else {
1453 SnapManager &m = _desktop->namedview->snap_manager;
1454 m.setup(_desktop, false, _objects_const);
1455 // When rotating, we cannot snap the corners of the bounding box, see the comment in "constrainedSnapRotate" for details
1456 Inkscape::PureRotateConstrained sn = Inkscape::PureRotateConstrained(radians, _origin);
1457 m.snapTransformed(_snap_points, _point, sn);
1458 m.unSetup();
1459
1460 if (sn.best_snapped_point.getSnapped()) {
1461 _desktop->snapindicator->set_new_snaptarget(sn.best_snapped_point);
1462 // We snapped something, so change the rotation to reflect it
1463 radians = sn.getAngleSnapped();
1464 r1 = Geom::Rotate(0);
1465 r2 = Geom::Rotate(radians);
1466 } else {
1467 _desktop->snapindicator->remove_snaptarget();
1468 }
1469
1470 }
1471
1472
1473 // Calculate the relative affine
1474 _relative_affine = r2 * r1.inverse();
1475
1476 // Update the handle position
1477 pt = _point * Geom::Translate(-_origin) * _relative_affine * Geom::Translate(_origin);
1478
1479 // Update the status text
