546 // Join the tokens in the given list using delimiter. If prefix is defined then each token
547 // will be concatenated with the prefix before being added to the composite string.
548 template <class T>
549 std::string
550 km_join(const T& list, const std::string& delimiter, const std::string& prefix = "")
551 {
552 std::string result;
553
554 for ( typename T::const_iterator i = list.begin(); i != list.end(); ++i )
555 {
556 if ( i != list.begin() )
557 {
558 result += delimiter;
559 }
560
561 result += prefix + *i;
562 }
563
564 return result;
565 }
